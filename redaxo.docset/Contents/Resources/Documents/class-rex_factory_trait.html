Trait rex_factory_trait

Factory trait.

Example child class:

class rex_example
{
    use rex_factory;

    private function __construct($param)
    {
        // ...
    }

    public static function factory($param)
    {
        $class = self::getFactoryClass();
        return new $class($param);
    }
}
