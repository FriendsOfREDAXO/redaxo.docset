  3 /**
  4  * Class for sockets.
  5  *
  6  * Example:
  7  * <code>
  8  *     try {
  9  *         $socket = rex_socket::factory('www.example.com');
 10  *         $socket->setPath('/url/to/my/resource?param=1');
 11  *         $response = $socket->doGet();
 12  *         if($response->isOk()) {
 13  *             $body = $response->getBody();
 14  *         }
 15  *     } catch(rex_socket_exception $e) {
 16  *         // error message: $e->getMessage()
 17  *     }
 18  * </code>
 19  *
 20  * @author gharlan
 21  *
 22  * @package redaxo\core
 23  */

 35     /**
 36      * Constructor.
 37      *
 38      * @param string $host Host name
 39      * @param int    $port Port number
 40      * @param bool   $ssl  SSL flag
 41      */
 42     protected function __construct($host, $port = 80, $ssl = false)
 43     {
 44         $this->host = $host;
 45         $this->port = $port;
 46         $this->ssl = $ssl;
 47 
 48         $this->addHeader('Host', $this->host);
 49         $this->addHeader('User-Agent', 'REDAXO/' . rex::getVersion());
 50         $this->addHeader('Connection', 'Close');
 51     }

 53     /**
 54      * Factory method.
 55      *
 56      * @param string $host Host name
 57      * @param int    $port Port number
 58      * @param bool   $ssl  SSL flag
 59      *
 60      * @return static Socket instance
 61      *
 62      * @see rex_socket::factoryUrl()
 63      */
 64     public static function factory($host, $port = 80, $ssl = false)
 65     {
 66         if (get_called_class() === __CLASS__ && ($proxy = rex::getProperty('socket_proxy'))) {
 67             return rex_socket_proxy::factoryUrl($proxy)->setDestination($host, $port, $ssl);
 68         }
 69 
 70         return new static($host, $port, $ssl);
 71     }

181     /**
182      * Makes a POST request.
183      *
184      * @param string|array|callable $data  Body data as string or array (POST parameters) or a callback for writing the body
185      * @param array                 $files Files array, e.g. `array('myfile' => array('path' => $path, 'type' => 'image/png'))`
186      *
187      * @return rex_socket_response Response
188      *
189      * @throws rex_socket_exception
190      */
191     public function doPost($data = '', array $files = [])
192     {
193         if (is_array($data) && !empty($files)) {
194             $data = function ($stream) use ($data, $files) {
195                 $boundary = '----------6n2Yd9bk2liD6piRHb5xF6';
196                 $eol = "\r\n";
197                 fwrite($stream, 'Content-Type: multipart/form-data; boundary=' . $boundary . $eol);
198                 $dataFormat = '--' . $boundary . $eol . 'Content-Disposition: form-data; name="%s"' . $eol . $eol;
199                 $fileFormat = '--' . $boundary . $eol . 'Content-Disposition: form-data; name="%s"; filename="%s"' . $eol . 'Content-Type: %s' . $eol . $eol;
200                 $end = '--' . $boundary . '--' . $eol;
201                 $length = 0;
202                 $temp = explode('&', rex_string::buildQuery($data));
203                 $data = [];
204                 $partLength = rex_string::size(sprintf($dataFormat, '') . $eol);
205                 foreach ($temp as $t) {
206                     list($key, $value) = array_map('urldecode', explode('=', $t, 2));
207                     $data[$key] = $value;
208                     $length += $partLength + rex_string::size($key) + rex_string::size($value);
209                 }
210                 $partLength = rex_string::size(sprintf($fileFormat, '', '', '') . $eol);
211                 foreach ($files as $key => $file) {
212                     $length += $partLength + rex_string::size($key) + rex_string::size(basename($file['path'])) + rex_string::size($file['type']) + filesize($file['path']);
213                 }
214                 $length += rex_string::size($end);
215                 fwrite($stream, 'Content-Length: ' . $length . $eol . $eol);
216                 foreach ($data as $key => $value) {
217                     fwrite($stream, sprintf($dataFormat, $key) . $value . $eol);
218                 }
219                 foreach ($files as $key => $file) {
220                     fwrite($stream, sprintf($fileFormat, $key, basename($file['path']), $file['type']));
221                     $file = fopen($file['path'], 'rb');
222                     while (!feof($file)) {
223                         fwrite($stream, fread($file, 1024));
224                     }
225                     fclose($file);
226                     fwrite($stream, $eol);
227                 }
228                 fwrite($stream, $end);
229             };
230         } elseif (!is_callable($data)) {
231             if (is_array($data)) {
232                 $data = rex_string::buildQuery($data);
233                 $this->addHeader('Content-Type', 'application/x-www-form-urlencoded');
234             }
235         }
236         return $this->doRequest('POST', $data);
237     }

251     /**
252      * Makes a request.
253      *
254      * @param string          $method HTTP method, e.g. "GET"
255      * @param string|callable $data   Body data as string or a callback for writing the body
256      *
257      * @return rex_socket_response Response
258      *
259      * @throws InvalidArgumentException
260      */
261     public function doRequest($method, $data = '')
262     {
263         if (!is_string($data) && !is_callable($data)) {
264             throw new InvalidArgumentException(sprintf('Expecting $data to be a string or a callable, but %s given!', gettype($data)));
265         }
266 
267         if (!$this->ssl) {
268             rex_logger::logError(E_WARNING, 'You should not use non-secure socket connections while connecting to "'. $this->host .'"!', __FILE__, __LINE__);
269         }
270 
271         $this->openConnection();
272         $response = $this->writeRequest($method, $this->path, $this->headers, $data);
273 
274         if ('GET' !== $method || !$this->followRedirects || !$response->isRedirection()) {
275             return $response;
276         }
277 
278         $location = $response->getHeader('location');
279 
280         if (!$location) {
281             return $response;
282         }
283 
284         if (false === strpos($location, '//')) {
285             $socket = self::factory($this->host, $this->port, $this->ssl)->setPath($location);
286         } else {
287             $socket = self::factoryUrl($location);
288 
289             if ($this->ssl && !$socket->ssl) {
290                 return $response;
291             }
292         }
293 
294         $socket->setTimeout($this->timeout);
295         $socket->followRedirects($this->followRedirects - 1);
296 
297         foreach ($this->headers as $key => $value) {
298             if ('Host' !== $key) {
299                 $socket->addHeader($key, $value);
300             }
301         }
302 
303         return $socket->doGet();
304     }

306     /**
307      * Opens the socket connection.
308      *
309      * @throws rex_socket_exception
310      */
311     protected function openConnection()
312     {
313         $host = ($this->ssl ? 'ssl://' : '') . $this->host;
314 
315         $prevError = null;
316         set_error_handler(function ($errno, $errstr) use (&$prevError) {
317             if (null === $prevError) {
318                 $prevError = $errstr;
319             }
320         });
321 
322         try {
323             $this->stream = @fsockopen($host, $this->port, $errno, $errstr);
324         } finally {
325             restore_error_handler();
326         }
327 
328         if ($this->stream) {
329             stream_set_timeout($this->stream, $this->timeout);
330 
331             return;
332         }
333 
334         if ($errstr) {
335             throw new rex_socket_exception($errstr . ' (' . $errno . ')');
336         }
337 
338         if ($prevError) {
339             throw new rex_socket_exception($prevError);
340         }
341 
342         throw new rex_socket_exception('Unknown error.');
343     }

345     /**
346      * Writes a request to the opened connection.
347      *
348      * @param string          $method  HTTP method, e.g. "GET"
349      * @param string          $path    Path
350      * @param array           $headers Headers
351      * @param string|callable $data    Body data as string or a callback for writing the body
352      *
353      * @throws rex_socket_exception
354      *
355      * @return rex_socket_response Response
356      */
357     protected function writeRequest($method, $path, array $headers = [], $data = '')
358     {
359         $eol = "\r\n";
360         $headerStrings = [];
361         $headerStrings[] = strtoupper($method) . ' ' . $path . ' HTTP/1.1';
362         foreach ($headers as $key => $value) {
363             $headerStrings[] = $key . ': ' . $value;
364         }
365         foreach ($headerStrings as $header) {
366             fwrite($this->stream, str_replace(["\r", "\n"], '', $header) . $eol);
367         }
368         if (!is_callable($data)) {
369             fwrite($this->stream, 'Content-Length: ' . rex_string::size($data) . $eol);
370             fwrite($this->stream, $eol . $data);
371         } else {
372             call_user_func($data, $this->stream);
373         }
374 
375         $meta = stream_get_meta_data($this->stream);
376         if (isset($meta['timed_out']) && $meta['timed_out']) {
377             throw new rex_socket_exception('Timeout!');
378         }
379 
380         return new rex_socket_response($this->stream);
381     }

383     /**
384      * Parses a full URL and returns an array with the keys "host", "port", "ssl" and "path".
385      *
386      * @param string $url Full URL
387      *
388      * @return array URL parts
389      *
390      * @throws rex_socket_exception
391      */
392     protected static function parseUrl($url)
393     {
394         $parts = parse_url($url);
395         if ($parts !== false && !isset($parts['host']) && strpos($url, 'http') !== 0) {
396             $parts = parse_url('http://' . $url);
397         }
398         if ($parts === false || !isset($parts['host'])) {
399             throw new rex_socket_exception('It isn\'t possible to parse the URL "' . $url . '"!');
400         }
401 
402         $port = 80;
403         $ssl = false;
404         if (isset($parts['scheme'])) {
405             $supportedProtocols = ['http', 'https'];
406             if (!in_array($parts['scheme'], $supportedProtocols)) {
407                 throw new rex_socket_exception('Unsupported protocol "' . $parts['scheme'] . '". Supported protocols are ' . implode(', ', $supportedProtocols) . '.');
408             }
409             if ($parts['scheme'] == 'https') {
410                 $ssl = true;
411                 $port = 443;
412             }
413         }
414         $port = isset($parts['port']) ? (int) $parts['port'] : $port;
415 
416         $path = (isset($parts['path']) ? $parts['path'] : '/')
417             . (isset($parts['query']) ? '?' . $parts['query'] : '')
418             . (isset($parts['fragment']) ? '#' . $parts['fragment'] : '');
419 
420         return [
421             'host' => $parts['host'],
422             'port' => $port,
423             'ssl' => $ssl,
424             'path' => $path,
425         ];
426     }
