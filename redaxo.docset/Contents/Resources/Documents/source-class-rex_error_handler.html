 10     /**
 11      * Registers the class as php-error/exception handler.
 12      */
 13     public static function register()
 14     {
 15         if (self::$registered) {
 16             return;
 17         }
 18 
 19         self::$registered = true;
 20 
 21         set_error_handler([__CLASS__, 'handleError']);
 22         set_exception_handler([__CLASS__, 'handleException']);
 23         register_shutdown_function([__CLASS__, 'shutdown']);
 24     }
 25 
 26     /**
 27      * Unregisters the logger instance as php-error/exception handler.
 28      */
 29     public static function unregister()
 30     {
 31         if (!self::$registered) {
 32             return;
 33         }
 34 
 35         self::$registered = false;
 36 
 37         restore_error_handler();
 38         restore_exception_handler();
 39         // unregister of shutdown function is not possible
 40     }

155     /**
156      * Handles a error message.
157      *
158      * @param int    $errno   The error code to handle
159      * @param string $errstr  The error message
160      * @param string $errfile The file in which the error occured
161      * @param int    $errline The line of the file in which the error occured
162      *
163      * @throws ErrorException
164      */
165     public static function handleError($errno, $errstr, $errfile, $errline)
166     {
167         if (in_array($errno, [E_USER_ERROR, E_ERROR, E_COMPILE_ERROR, E_RECOVERABLE_ERROR, E_PARSE])) {
168             throw new ErrorException($errstr, 0, $errno, $errfile, $errline);
169         }
170 
171         // silenced errors ("@" operator)
172         if (0 === error_reporting()) {
173             return;
174         }
175 
176         $debug = rex::getDebugFlags();
177 
178         if (
179             isset($debug['throw_always_exception']) &&
180             (true === $debug['throw_always_exception'] || $errno === ($errno & $debug['throw_always_exception']))
181         ) {
182             throw new ErrorException($errstr, 0, $errno, $errfile, $errline);
183         }
184 
185         if ((error_reporting() & $errno) !== $errno) {
186             return;
187         }
188 
189         if (ini_get('display_errors') && (rex::isSetup() || rex::isDebugMode() || ($user = rex_backend_login::createUser()) && $user->isAdmin())) {
190             echo '<div><b>' . self::getErrorType($errno) . "</b>: $errstr in <b>$errfile</b> on line <b>$errline</b></div>";
191         }
192 
193         rex_logger::logError($errno, $errstr, $errfile, $errline);
194     }

196     /**
197      * Shutdown-handler which is called at the very end of the request.
198      */
199     public static function shutdown()
200     {
201         // catch fatal/parse errors
202         if (self::$registered) {
203             $error = error_get_last();
204             if (is_array($error) && in_array($error['type'], [E_USER_ERROR, E_ERROR, E_COMPILE_ERROR, E_RECOVERABLE_ERROR, E_PARSE])) {
205                 self::handleException(new ErrorException($error['message'], 0, $error['type'], $error['file'], $error['line']));
206             }
207         }
208     }

210     /**
211      * Get a human readable string representing the given php error code.
212      *
213      * @param int $errno a php error code, e.g. E_ERROR
214      *
215      * @return string
216      */
217     public static function getErrorType($errno)
218     {
219         switch ($errno) {
220             case E_USER_ERROR:
221             case E_ERROR:
222             case E_COMPILE_ERROR:
223             case E_RECOVERABLE_ERROR:
224                 return 'Fatal error';
225 
226             case E_PARSE:
227                 return 'Parse error';
228 
229             case E_USER_WARNING:
230             case E_WARNING:
231             case E_COMPILE_WARNING:
232                 return 'Warning';
233 
234             case E_USER_NOTICE:
235             case E_NOTICE:
236                 return 'Notice';
237 
238             case E_USER_DEPRECATED:
239             case E_DEPRECATED:
240                 return 'Deprecated';
241 
242             case E_STRICT:
243                 return 'Strict';
244 
245             default:
246                 return 'Unknown';
247         }
248     }
