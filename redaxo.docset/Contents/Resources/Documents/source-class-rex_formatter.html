 18      * Formats a string by the given format type.
 19      *
 20      * @param string $value      Value
 21      * @param string $formatType Format type (any method name of this class)
 22      * @param mixed  $format     For possible values look at the other methods of this class
 23      *
 24      * @return string
 25      *
 26      * @throws InvalidArgumentException
 27      */
 28     public static function format($value, $formatType, $format)
 29     {
 30         if (!is_callable([__CLASS__, $formatType])) {
 31             throw new InvalidArgumentException('Unknown $formatType: "' . $formatType . '"!');
 32         }
 33         return self::$formatType($value, $format);
 34     }

 36     /**
 37      * Formats a string by `date()`.
 38      *
 39      * @see http://www.php.net/manual/en/function.date.php
 40      *
 41      * @param string $value  Unix timestamp or datetime string for `strtotime`
 42      * @param string $format Default format is `d.m.Y`
 43      *
 44      * @return string
 45      */
 46     public static function date($value, $format = '')
 47     {
 48         if (empty($value)) {
 49             return '';
 50         }
 51 
 52         if ($format == '') {
 53             $format = 'd.m.Y';
 54         }
 55 
 56         return date($format, self::getTimestamp($value));
 57     }

 59     /**
 60      * Formats a string by `strftime()`.
 61      *
 62      * @see http://www.php.net/manual/en/function.strftime.php
 63      *
 64      * @param string $value  Unix timestamp or datetime string for `strtotime`
 65      * @param string $format Possible values are format strings like in `strftime` or "date" or "datetime", default is "date"
 66      *
 67      * @return string
 68      */
 69     public static function strftime($value, $format = '')
 70     {
 71         if (empty($value)) {
 72             return '';
 73         }
 74 
 75         if ($format == '' || $format == 'date') {
 76             // Default REX-Dateformat
 77             $format = rex_i18n::msg('dateformat');
 78         } elseif ($format == 'datetime') {
 79             // Default REX-Datetimeformat
 80             $format = rex_i18n::msg('datetimeformat');
 81         }
 82         return strftime($format, self::getTimestamp($value));
 83     }

 85     /**
 86      * Formats a string by `number_format()`.
 87      *
 88      * @see http://www.php.net/manual/en/function.number-format.php
 89      *
 90      * @param string $value  Value
 91      * @param array  $format Array with number of decimals, decimals point and thousands separator, default is `array(2, ',', ' ')`
 92      *
 93      * @return string
 94      */
 95     public static function number($value, $format = [])
 96     {
 97         if (!is_array($format)) {
 98             $format = [];
 99         }
100 
101         // Kommastellen
102         if (!isset($format[0])) {
103             $format[0] = 2;
104         }
105         // Dezimal Trennzeichen
106         if (!isset($format[1])) {
107             $format[1] = ',';
108         }
109         // Tausender Trennzeichen
110         if (!isset($format[2])) {
111             $format[2] = ' ';
112         }
113         return number_format($value, $format[0], $format[1], $format[2]);
114     }

116     /**
117      * Formats a string as bytes.
118      *
119      * @param string $value  Value
120      * @param array  $format Same as {@link rex_formatter::number()}
121      *
122      * @return string
123      */
124     public static function bytes($value, $format = [])
125     {
126         $units = ['B', 'KiB', 'MiB', 'GiB', 'TiB', 'PiB', 'EiB', 'ZiB', 'YiB'];
127         $unit_index = 0;
128         while (($value / 1024) >= 1) {
129             $value /= 1024;
130             ++$unit_index;
131         }
132 
133         if (isset($format[0])) {
134             $z = (int) ($value * pow(10, $precision = (int) ($format[0])));
135             for ($i = 0; $i < (int) $precision; ++$i) {
136                 if (($z % 10) == 0) {
137                     $format[0] = (int) ($format[0]) - 1;
138                     $z = (int) ($z / 10);
139                 } else {
140                     break;
141                 }
142             }
143         }
144 
145         return self::number($value, $format) . ' ' . $units[$unit_index];
146     }

180     /**
181      * Truncates a string.
182      *
183      * @param string $value  Value
184      * @param array  $format Default format is `array('length' => 80, 'etc' => '...', 'break_words' => false)`
185      *
186      * @return string
187      */
188     public static function truncate($value, $format = [])
189     {
190         if (!is_array($format)) {
191             $format = [];
192         }
193 
194         // Max-String-laenge
195         if (empty($format['length'])) {
196             $format['length'] = 80;
197         }
198 
199         // ETC
200         if (empty($format['etc'])) {
201             $format['etc'] = '...';
202         }
203 
204         // Break-Words?
205         if (empty($format['break_words'])) {
206             $format['break_words'] = false;
207         }
208 
209         if (mb_strlen($value) > $format['length']) {
210             $format['length'] -= mb_strlen($format['etc']);
211             if (!$format['break_words']) {
212                 $value = preg_replace('/\s+?(\S+)?$/', '', substr($value, 0, $format['length'] + 1));
213             }
214 
215             return substr($value, 0, $format['length']) . $format['etc'];
216         }
217 
218         return $value;
219     }

221     /**
222      * Avoid widows in a string.
223      *
224      * @param string $value
225      *
226      * @return string
227      */
228     public static function widont($value)
229     {
230         // Sollte ein Wort allein auf einer Zeile vorkommen, wird dies unterbunden
231         $value = rtrim($value);
232         $space = strrpos($value, ' ');
233         if ($space !== false) {
234             $value = substr($value, 0, $space) . '&#160;' . substr($value, $space + 1);
235         }
236         return $value;
237     }

254     /**
255      * Formats a string as link.
256      *
257      * @param string $value  URL
258      * @param array  $format Array with link attributes and params
259      *
260      * @return string Link
261      */
262     public static function url($value, $format = [])
263     {
264         if (empty($value)) {
265             return '';
266         }
267 
268         if (!is_array($format)) {
269             $format = [];
270         }
271 
272         // Linkattribute
273         if (empty($format['attr'])) {
274             $format['attr'] = '';
275         }
276         // Linkparameter (z.b. subject=Hallo Sir)
277         if (empty($format['params'])) {
278             $format['params'] = '';
279         } else {
280             if (strstr($format['params'], '?') != $format['params']) {
281                 $format['params'] = '?' . $format['params'];
282             }
283         }
284         // Protokoll
285         if (!preg_match('@((ht|f)tps?|telnet|redaxo)://@', $value)) {
286             $value = 'http://' . $value;
287         }
288 
289         return '<a href="' . htmlspecialchars($value . $format['params']) . '"' . $format['attr'] . '>' . htmlspecialchars($value) . '</a>';
290     }

322     /**
323      * Formats a string by a custom callable.
324      *
325      * @param string         $value  Value
326      * @param callable|array $format A callable or an array of a callable and additional params
327      *
328      * @return string
329      *
330      * @throws rex_exception
331      */
332     public static function custom($value, $format)
333     {
334         if (!is_callable($format)) {
335             if (!is_callable($format[0])) {
336                 throw new rex_exception('Unable to find callable ' . $format[0] . ' for custom format!');
337             }
338 
339             $params = [];
340             $params['subject'] = $value;
341             if (is_array($format[1])) {
342                 $params = array_merge($format[1], $params);
343             } else {
344                 $params['params'] = $format[1];
345             }
346             // $format ist in der Form
347             // array(Name des Callables, Weitere Parameter)
348             return call_user_func($format[0], $params);
349         }
350 
351         return call_user_func($format, $value);
352     }

354     private static function getTimestamp($value)
355     {
356         if (is_numeric($value)) {
357             return $value;
358         }
359 
360         return strtotime($value);
