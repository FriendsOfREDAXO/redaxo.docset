 3 /**
 4  * Markdown parser.
 5  *
 6  * @author gharlan
 7  *
 8  * @package redaxo\core
 9  */
10 class rex_markdown
11 {
12     use rex_factory_trait;
13 
14     /** @var ParsedownExtra */
15     private $parser;
16 
17     private function __construct()
18     {
19         $this->parser = new ParsedownExtra();
20         $this->parser->setBreaksEnabled(true);
21     }
22 
23     /**
24      * @return static
25      */
26     public static function factory()
27     {
28         $class = static::getFactoryClass();
29         return new $class();
30     }
31 
32     /**
33      * Parses markdown code.
34      *
35      * @param string $code Markdown code
36      *
37      * @return string HTML code
38      */
39     public function parse($code)
40     {
41         return $this->parser->text($code);
42     }
43 }
