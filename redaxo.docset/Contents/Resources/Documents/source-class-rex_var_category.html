21     protected function getOutput()
22     {
23         $field = $this->getParsedArg('field');
24         if (!$field) {
25             return false;
26         }
27 
28         $category_id = $this->getParsedArg('id', '$this->getValue(\'category_id\')');
29         $clang = $this->getParsedArg('clang', 'null');
30 
31         return __CLASS__ . '::getCategoryValue(' . $category_id . ', ' . $field . ', ' . $clang . ')';
32     }
33 
34     public static function getCategoryValue($id, $field, $clang = null)
35     {
36         if ($clang === null) {
37             $clang = rex_clang::getCurrentId();
38         }
39         $cat = rex_category::get($id, $clang);
40         if ($cat) {
41             return htmlspecialchars($cat->getValue($field));
42         }
