  3 /**
  4  * Class for user roles.
  5  *
  6  * @author gharlan
  7  *
  8  * @package redaxo\users
  9  */
 10 class rex_user_role implements rex_user_role_interface
 11 {
 12     /**
 13      * Permissions.
 14      *
 15      * @var array
 16      */
 17     private $perms = [];
 18 
 19     /**
 20      * Complex perm params.
 21      *
 22      * @var array
 23      */
 24     private $complexPermParams = [];
 25 
 26     /**
 27      * Cache for complex perm instances.
 28      *
 29      * @var array
 30      */
 31     private $complexPerms = [];
 32 
 33     /**
 34      * Constructor.
 35      *
 36      * @param array $roles
 37      */
 38     private function __construct(array $roles)
 39     {
 40         foreach ($roles as $role) {
 41             foreach ([rex_perm::GENERAL, rex_perm::OPTIONS, rex_perm::EXTRAS] as $key) {
 42                 $perms = $role[$key] ? explode('|', trim($role[$key], '|')) : [];
 43                 $this->perms = array_merge($this->perms, $perms);
 44                 unset($role[$key]);
 45             }
 46 
 47             foreach ($role as $key => $value) {
 48                 $perms = $role[$key] == rex_complex_perm::ALL ? rex_complex_perm::ALL : explode('|', trim($role[$key], '|'));
 49                 if (count($perms) == 1 && $perms[0] == '') {
 50                     $perms = [];
 51                 }
 52 
 53                 if (!isset($this->complexPermParams[$key])) {
 54                     $this->complexPermParams[$key] = $perms;
 55                 } elseif ($this->complexPermParams[$key] == rex_complex_perm::ALL) {
 56                 } elseif ($perms == rex_complex_perm::ALL) {
 57                     $this->complexPermParams[$key] = $perms;
 58                 } else {
 59                     $this->complexPermParams[$key] = array_merge($perms, $this->complexPermParams[$key]);
 60                 }
 61             }
 62         }
 63     }
 64 
 65     /**
 66      * {@inheritdoc}
 67      */
 68     public function hasPerm($perm)
 69     {
 70         return in_array($perm, $this->perms);
 71     }
 72 
 73     /**
 74      * {@inheritdoc}
 75      */
 76     public function getComplexPerm(rex_user $user, $key)
 77     {
 78         if (isset($this->complexPerms[$key])) {
 79             return $this->complexPerms[$key];
 80         }
 81         if (!isset($this->complexPermParams[$key])) {
 82             $this->complexPermParams[$key] = [];
 83         }
 84         $this->complexPerms[$key] = rex_complex_perm::get($user, $key, $this->complexPermParams[$key]);
 85         return $this->complexPerms[$key];
 86     }
 87 
 88     /**
 89      * {@inheritdoc}
 90      */
 91     public static function get($ids)
 92     {
 93         $sql = rex_sql::factory();
 94         $user_roles = $sql->getArray('SELECT perms FROM ' . rex::getTablePrefix() . 'user_role WHERE FIND_IN_SET(id, ?)', [$ids]);
 95         if (count($user_roles) == 0) {
 96             return null;
 97         }
 98 
 99         $roles = [];
100         foreach ($user_roles as $user_role) {
101             $roles[] = json_decode($user_role['perms'], true);
102         }
103 
104         return new self($roles);
105     }
106 
107     public static function removeOrReplaceItem(rex_extension_point $ep)
108     {
109         $params = $ep->getParams();
110         $key = $params['key'];
111         $item = '|' . $params['item'] . '|';
112         $new = isset($params['new']) ? '|' . $params['new'] . '|' : '|';
113         $sql = rex_sql::factory();
114         $sql->setQuery('SELECT id, perms FROM ' . rex::getTable('user_role'));
115         $update = rex_sql::factory();
116         $update->prepareQuery('UPDATE ' . rex::getTable('user_role') . ' SET perms = ? WHERE id = ?');
117         foreach ($sql as $row) {
118             $perms = json_decode($row->getValue('perms'), true);
119             if (isset($perms[$key]) && strpos($perms[$key], $item) !== false) {
120                 $perms[$key] = str_replace($item, $new, $perms[$key]);
121                 $update->execute([json_encode($perms), $row->getValue('id')]);
122             }
123         }
124     }
125 }
