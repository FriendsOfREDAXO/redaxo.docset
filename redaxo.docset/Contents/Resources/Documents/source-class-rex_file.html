  3 /**
  4  * Class for handling files.
  5  *
  6  * @package redaxo\core
  7  */
  8 class rex_file
  9 {
 10     /**
 11      * Returns the content of a file.
 12      *
 13      * @param string $file    Path to the file
 14      * @param mixed  $default Default value
 15      *
 16      * @return mixed Content of the file or default value if the file isn't readable
 17      */
 18     public static function get($file, $default = null)
 19     {
 20         return rex_timer::measure(__METHOD__, function () use ($file, $default) {
 21             $content = @file_get_contents($file);
 22             return $content !== false ? $content : $default;
 23         });
 24     }
 25 
 26     /**
 27      * Returns the content of a config file.
 28      *
 29      * @param string $file    Path to the file
 30      * @param mixed  $default Default value
 31      *
 32      * @return mixed Content of the file or default value if the file isn't readable
 33      */
 34     public static function getConfig($file, $default = [])
 35     {
 36         $content = self::get($file);
 37         return $content === null ? $default : rex_string::yamlDecode($content);
 38     }
 39 
 40     /**
 41      * Returns the content of a cache file.
 42      *
 43      * @param string $file    Path to the file
 44      * @param mixed  $default Default value
 45      *
 46      * @return mixed Content of the file or default value if the file isn't readable
 47      */
 48     public static function getCache($file, $default = [])
 49     {
 50         $content = self::get($file);
 51         return $content === null ? $default : json_decode($content, true);
 52     }
 53 
 54     /**
 55      * Puts content in a file.
 56      *
 57      * @param string $file    Path to the file
 58      * @param string $content Content for the file
 59      *
 60      * @return bool TRUE on success, FALSE on failure
 61      */
 62     public static function put($file, $content)
 63     {
 64         return rex_timer::measure(__METHOD__, function () use ($file, $content) {
 65             if (!rex_dir::create(dirname($file)) || file_exists($file) && !is_writable($file)) {
 66                 return false;
 67             }
 68 
 69             if (file_put_contents($file, $content) !== false) {
 70                 @chmod($file, rex::getFilePerm());
 71                 return true;
 72             }
 73 
 74             return false;
 75         });
 76     }
 77 
 78     /**
 79      * Puts content in a config file.
 80      *
 81      * @param string $file    Path to the file
 82      * @param mixed  $content Content for the file
 83      * @param int    $inline  The level where you switch to inline YAML
 84      *
 85      * @return bool TRUE on success, FALSE on failure
 86      */
 87     public static function putConfig($file, $content, $inline = 3)
 88     {
 89         return self::put($file, rex_string::yamlEncode($content, $inline));
 90     }
 91 
 92     /**
 93      * Puts content in a cache file.
 94      *
 95      * @param string $file    Path to the file
 96      * @param mixed  $content Content for the file
 97      *
 98      * @return bool TRUE on success, FALSE on failure
 99      */
100     public static function putCache($file, $content)
101     {
102         return self::put($file, json_encode($content));
103     }
104 
105     /**
106      * Copies a file.
107      *
108      * @param string $srcfile Path of the source file
109      * @param string $dstfile Path of the destination file or directory
110      *
111      * @return bool TRUE on success, FALSE on failure
112      */
113     public static function copy($srcfile, $dstfile)
114     {
115         return rex_timer::measure(__METHOD__, function () use ($srcfile, $dstfile) {
116             if (is_file($srcfile)) {
117                 if (is_dir($dstfile)) {
118                     $dstdir = rtrim($dstfile, DIRECTORY_SEPARATOR);
119                     $dstfile = $dstdir . DIRECTORY_SEPARATOR . basename($srcfile);
120                 } else {
121                     $dstdir = dirname($dstfile);
122                     rex_dir::create($dstdir);
123                 }
124 
125                 if (rex_dir::isWritable($dstdir) && (!file_exists($dstfile) || is_writable($dstfile)) && copy($srcfile, $dstfile)) {
126                     @chmod($dstfile, rex::getFilePerm());
127                     touch($dstfile, filemtime($srcfile), fileatime($srcfile));
128                     return true;
129                 }
130             }
131             return false;
132         });
133     }
134 
135     /**
136      * Deletes a file.
137      *
138      * @param string $file Path of the file
139      *
140      * @return bool TRUE on success, FALSE on failure
141      */
142     public static function delete($file)
143     {
144         return rex_timer::measure(__METHOD__, function () use ($file) {
145             if (file_exists($file)) {
146                 return unlink($file);
147             }
148             return true;
149         });
150     }
151 
152     /**
153      * Extracts the extension of the given filename.
154      *
155      * @param string $filename Filename
156      *
157      * @return string Extension of $filename
158      */
159     public static function extension($filename)
160     {
161         return pathinfo($filename, PATHINFO_EXTENSION);
162     }
163 
164     /**
165      * Formates the filesize of the given file into a userfriendly form.
166      *
167      * @param string $file   Path to the file
168      * @param array  $format
169      *
170      * @return string Formatted filesize
171      */
172     public static function formattedSize($file, $format = [])
173     {
174         return rex_formatter::bytes(filesize($file), $format);
175     }
176 
177     /**
178      * Gets executed content of given file.
179      *
180      * @param string $file Path of the file
181      *
182      * @return string executed Content
183      */
184     public static function getOutput($file)
185     {
186         ob_start();
187         require $file;
188         return ob_get_clean();
189     }
