  3 /**
  4  * Class for handling files.
  5  *
  6  * @package redaxo\core
  7  */
  8 class rex_file
  9 {
 10     /**
 11      * Returns the content of a file.
 12      *
 13      * @param string $file    Path to the file
 14      * @param mixed  $default Default value
 15      *
 16      * @return mixed Content of the file or default value if the file isn't readable
 17      */
 18     public static function get($file, $default = null)
 19     {
 20         return is_readable($file) ? file_get_contents($file) : $default;
 21     }
 22 
 23     /**
 24      * Returns the content of a config file.
 25      *
 26      * @param string $file    Path to the file
 27      * @param mixed  $default Default value
 28      *
 29      * @return mixed Content of the file or default value if the file isn't readable
 30      */
 31     public static function getConfig($file, $default = [])
 32     {
 33         $content = self::get($file);
 34         return $content === null ? $default : rex_string::yamlDecode($content);
 35     }
 36 
 37     /**
 38      * Returns the content of a cache file.
 39      *
 40      * @param string $file    Path to the file
 41      * @param mixed  $default Default value
 42      *
 43      * @return mixed Content of the file or default value if the file isn't readable
 44      */
 45     public static function getCache($file, $default = [])
 46     {
 47         $content = self::get($file);
 48         return $content === null ? $default : json_decode($content, true);
 49     }
 50 
 51     /**
 52      * Puts content in a file.
 53      *
 54      * @param string $file    Path to the file
 55      * @param string $content Content for the file
 56      *
 57      * @return bool TRUE on success, FALSE on failure
 58      */
 59     public static function put($file, $content)
 60     {
 61         if (!rex_dir::create(dirname($file)) || file_exists($file) && !is_writable($file)) {
 62             return false;
 63         }
 64 
 65         if (file_put_contents($file, $content) !== false) {
 66             @chmod($file, rex::getFilePerm());
 67             return true;
 68         }
 69 
 70         return false;
 71     }
 72 
 73     /**
 74      * Puts content in a config file.
 75      *
 76      * @param string $file    Path to the file
 77      * @param mixed  $content Content for the file
 78      * @param int    $inline  The level where you switch to inline YAML
 79      *
 80      * @return bool TRUE on success, FALSE on failure
 81      */
 82     public static function putConfig($file, $content, $inline = 3)
 83     {
 84         return self::put($file, rex_string::yamlEncode($content, $inline));
 85     }
 86 
 87     /**
 88      * Puts content in a cache file.
 89      *
 90      * @param string $file    Path to the file
 91      * @param mixed  $content Content for the file
 92      *
 93      * @return bool TRUE on success, FALSE on failure
 94      */
 95     public static function putCache($file, $content)
 96     {
 97         return self::put($file, json_encode($content));
 98     }
 99 
100     /**
101      * Copies a file.
102      *
103      * @param string $srcfile Path of the source file
104      * @param string $dstfile Path of the destination file or directory
105      *
106      * @return bool TRUE on success, FALSE on failure
107      */
108     public static function copy($srcfile, $dstfile)
109     {
110         if (is_file($srcfile)) {
111             if (is_dir($dstfile)) {
112                 $dstdir = rtrim($dstfile, DIRECTORY_SEPARATOR);
113                 $dstfile = $dstdir . DIRECTORY_SEPARATOR . basename($srcfile);
114             } else {
115                 $dstdir = dirname($dstfile);
116                 rex_dir::create($dstdir);
117             }
118 
119             if (rex_dir::isWritable($dstdir) && (!file_exists($dstfile) || is_writable($dstfile)) && copy($srcfile, $dstfile)) {
120                 @chmod($dstfile, rex::getFilePerm());
121                 touch($dstfile, filemtime($srcfile));
122                 return true;
123             }
124         }
125         return false;
126     }
127 
128     /**
129      * Deletes a file.
130      *
131      * @param string $file Path of the file
132      *
133      * @return bool TRUE on success, FALSE on failure
134      */
135     public static function delete($file)
136     {
137         if (file_exists($file)) {
138             return unlink($file);
139         }
140         return true;
141     }
142 
143     /**
144      * Extracts the extension of the given filename.
145      *
146      * @param string $filename Filename
147      *
148      * @return string Extension of $filename
149      */
150     public static function extension($filename)
151     {
152         return pathinfo($filename, PATHINFO_EXTENSION);
153     }
154 
155     /**
156      * Formates the filesize of the given file into a userfriendly form.
157      *
158      * @param string $file   Path to the file
159      * @param array  $format
160      *
161      * @return string Formatted filesize
162      */
163     public static function formattedSize($file, $format = [])
164     {
165         return rex_formatter::bytes(filesize($file), $format);
166     }
167 
168     /**
169      * Gets executed content of given file.
170      *
171      * @param string $file Path of the file
172      *
173      * @return string executed Content
174      */
175     public static function getOutput($file)
176     {
177         ob_start();
178         require $file;
179         return ob_get_clean();
180     }
181 }
