1 <?php
  2 
  3 /**
  4  * @package redaxo\core\minibar
  5  */
  6 class rex_minibar
  7 {
  8     use rex_singleton_trait;
  9 
 10     /**
 11      * @var bool|null
 12      */
 13     private $isActive = null;
 14 
 15     /* @var rex_minibar_element[] */
 16     private $elements = [];
 17 
 18     public function addElement(rex_minibar_element $instance)
 19     {
 20         $this->elements[] = $instance;
 21     }
 22 
 23     /**
 24      * @param string $className
 25      *
 26      * @return rex_minibar_element|null
 27      */
 28     public function elementByClass($className)
 29     {
 30         foreach ($this->elements as $element) {
 31             if (get_class($element) === $className) {
 32                 return $element;
 33             }
 34         }
 35     }
 36 
 37     public function get()
 38     {
 39         if (!self::shouldRender()) {
 40             return null;
 41         }
 42 
 43         if (!count($this->elements)) {
 44             return null;
 45         }
 46 
 47         $fragment = new rex_fragment([
 48             'elements' => $this->elements,
 49         ]);
 50 
 51         if (rex::isBackend()) {
 52             return $fragment->parse('core/minibar/minibar_backend.php');
 53         }
 54 
 55         return $fragment->parse('core/minibar/minibar_frontend.php');
 56     }
 57 
 58     /**
 59      * Returns if the minibar should be rendered.
 60      *
 61      * @return bool
 62      */
 63     public function shouldRender()
 64     {
 65         if (is_bool($this->isActive)) {
 66             return $this->isActive;
 67         }
 68 
 69         $user = rex_backend_login::createUser();
 70         if (!$user) {
 71             return false;
 72         }
 73 
 74         if (rex::isBackend()) {
 75             return true;
 76         }
 77 
 78         return $user->getValue('minibar') == 1;
 79     }
 80 
 81     /**
 82      * Returns if the minibar is visible.
 83      *
 84      * @return bool
 85      */
 86     public function isVisible()
 87     {
 88         return !rex_cookie('rex_minibar_frontend_hidden', 'bool', false);
 89     }
 90 
 91     /**
 92      * Sets the visibility.
 93      *
 94      * @param bool $value
 95      */
 96     public function setVisibility($value)
 97     {
 98         if ($value) {
 99             rex_response::sendCookie('rex_minibar_frontend_hidden', '');
100         } else {
101             rex_response::sendCookie('rex_minibar_frontend_hidden', '1', ['expires' => time() + rex::getProperty('session_duration'), 'samesite' => 'strict']);
102         }
103     }
104 
105     /**
106      * @param bool $isActive
107      */
108     public function setActive($isActive)
109     {
110         $this->isActive = $isActive;
111     }
112 
113     /**
114      * @return bool|null
115      */
116     public function isActive()
117     {
118         return $this->isActive;
119     }
120 }
