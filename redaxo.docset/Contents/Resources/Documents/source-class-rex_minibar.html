 3 /**
 4  * @package redaxo\core
 5  */
 6 class rex_minibar
 7 {
 8     use rex_singleton_trait;
 9 
10     /* @var rex_minibar_element[] */
11     private $elements = [];
12 
13     public function addElement(rex_minibar_element $instance)
14     {
15         $this->elements[] = $instance;
16     }
17 
18     public function get()
19     {
20         if (!self::isActive()) {
21             return null;
22         }
23 
24         if (!count($this->elements)) {
25             return null;
26         }
27 
28         $fragment = new rex_fragment([
29             'elements' => $this->elements,
30         ]);
31 
32         if (rex::isBackend()) {
33             return $fragment->parse('core/minibar/minibar_backend.php');
34         }
35 
36         return $fragment->parse('core/minibar/minibar_frontend.php');
37     }
38 
39     /**
40      * Returns if the minibar is active.
41      *
42      * @return bool
43      */
44     public function isActive()
45     {
46         $user = rex_backend_login::createUser();
47         if (!$user) {
48             return false;
49         }
50 
51         if (rex::isBackend()) {
52             return true;
53         }
54 
55         return $user->getValue('minibar') == 1;
56     }
57 
58     /**
59      * Returns if the minibar is visible.
60      *
61      * @return bool
62      */
63     public function isVisible()
64     {
65         return !rex_cookie('rex_minibar_frontend_hidden', 'bool', false);
66     }
67 
68     /**
69      * Sets the visibility.
70      *
71      * @param bool $value
72      */
73     public function setVisibility($value)
74     {
75         if ($value) {
76             rex_response::sendCookie('rex_minibar_frontend_hidden', '');
77         } else {
78             rex_response::sendCookie('rex_minibar_frontend_hidden', '1', ['expires' => time() + rex::getProperty('session_duration'), 'samesite' => 'strict']);
79         }
80     }
81 }
