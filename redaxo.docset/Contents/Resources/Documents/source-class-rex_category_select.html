1 <?php

  3 /**
  4  * @package redaxo\structure
  5  */
  6 class rex_category_select extends rex_select
  7 {
  8     private $ignore_offlines;
  9     private $clang;
 10     private $check_perms;
 11 
 12     /**
 13      * @var int
 14      */
 15     private $rootId;
 16 
 17     private $loaded;
 18 
 19     public function __construct($ignore_offlines = false, $clang = false, $check_perms = true, $add_homepage = true)
 20     {
 21         $this->ignore_offlines = $ignore_offlines;
 22         $this->clang = $clang;
 23         $this->check_perms = $check_perms;
 24         $this->add_homepage = $add_homepage;
 25         $this->rootId = null;
 26         $this->loaded = false;
 27 
 28         parent::__construct();
 29     }
 30 
 31     /**
 32      * Kategorie-Id oder ein Array von Kategorie-Ids als Wurzelelemente der Select-Box.
 33      *
 34      * @param mixed $rootId Kategorie-Id oder Array von Kategorie-Ids zur Identifikation der Wurzelelemente
 35      */
 36     public function setRootId($rootId)
 37     {
 38         $this->rootId = $rootId;
 39     }
 40 
 41     protected function addCatOptions()
 42     {
 43         if ($this->add_homepage) {
 44             $this->addOption('Homepage', 0);
 45         }
 46 
 47         if ($this->rootId !== null) {
 48             if (is_array($this->rootId)) {
 49                 foreach ($this->rootId as $rootId) {
 50                     if ($rootCat = rex_category::get($rootId, $this->clang)) {
 51                         $this->addCatOption($rootCat, 0);
 52                     }
 53                 }
 54             } else {
 55                 if ($rootCat = rex_category::get($this->rootId, $this->clang)) {
 56                     $this->addCatOption($rootCat, 0);
 57                 }
 58             }
 59         } else {
 60             if (!$this->check_perms || rex::getUser()->getComplexPerm('structure')->hasCategoryPerm(0)) {
 61                 if ($rootCats = rex_category::getRootCategories($this->ignore_offlines, $this->clang)) {
 62                     foreach ($rootCats as $rootCat) {
 63                         $this->addCatOption($rootCat);
 64                     }
 65                 }
 66             } elseif (rex::getUser()->getComplexPerm('structure')->hasMountpoints()) {
 67                 $mountpoints = rex::getUser()->getComplexPerm('structure')->getMountpoints();
 68                 foreach ($mountpoints as $id) {
 69                     $cat = rex_category::get($id, $this->clang);
 70                     if ($cat && !rex::getUser()->getComplexPerm('structure')->hasCategoryPerm($cat->getParentId())) {
 71                         $this->addCatOption($cat, 0);
 72                     }
 73                 }
 74             }
 75         }
 76     }
 77 
 78     protected function addCatOption(rex_category $cat, $group = null)
 79     {
 80         if (!$this->check_perms ||
 81                 $this->check_perms && rex::getUser()->getComplexPerm('structure')->hasCategoryPerm($cat->getId(), false)
 82         ) {
 83             $cid = $cat->getId();
 84             $cname = $cat->getName() . ' [' . $cid . ']';
 85 
 86             if ($group === null) {
 87                 $group = $cat->getParentId();
 88             }
 89 
 90             $this->addOption($cname, $cid, $cid, $group);
 91             $childs = $cat->getChildren($this->ignore_offlines, $this->clang);
 92             if (is_array($childs)) {
 93                 foreach ($childs as $child) {
 94                     $this->addCatOption($child);
 95                 }
 96             }
 97         }
 98     }
 99 
100     public function get()
101     {
102         if (!$this->loaded) {
103             $this->addCatOptions();
104             $this->loaded = true;
105         }
106 
107         return parent::get();
108     }
109 
110     protected function outGroup($parent_id, $level = 0)
111     {
112         if ($level > 100) {
113             // nur mal so zu sicherheit .. man weiss nie ;)
114             throw new rex_exception('select->_outGroup overflow');
115         }
116 
117         $ausgabe = '';
118         $group = $this->getGroup($parent_id);
119         if (!is_array($group)) {
120             return '';
121         }
122         foreach ($group as $option) {
123             $name = $option[0];
124             $value = $option[1];
125             $id = $option[2];
126             if ($id == 0 || !$this->check_perms || ($this->check_perms && rex::getUser()->getComplexPerm('structure')->hasCategoryPerm($option[2]))) {
127                 $ausgabe .= $this->outOption($name, $value, $level);
128             } elseif (($this->check_perms && rex::getUser()->getComplexPerm('structure')->hasCategoryPerm($option[2]))) {
129                 --$level;
130             }
131 
132             $subgroup = $this->getGroup($id, true);
133             if ($subgroup !== false) {
134                 $ausgabe .= $this->outGroup($id, $level + 1);
135             }
136         }
137         return $ausgabe;
138     }
