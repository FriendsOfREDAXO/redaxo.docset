148     /**
149      * Prüft die mit setLogin() und setPassword() gesetzten Werte
150      * anhand des LoginQueries/UserQueries und gibt den Status zurück.
151      *
152      * Gibt true zurück bei erfolg, sonst false
153      */
154     public function checkLogin()
155     {
156         // wenn logout dann header schreiben und auf error seite verweisen
157         // message schreiben
158 
159         $ok = false;
160 
161         if (!$this->logout) {
162             // LoginStatus: 0 = noch checken, 1 = ok, -1 = not ok
163 
164             // gecachte ausgabe erlaubt ? checkLogin schonmal ausgeführt ?
165             if ($this->cache && $this->loginStatus != 0) {
166                 return $this->loginStatus > 0;
167             }
168 
169             if ($this->userLogin != '') {
170                 // wenn login daten eingegeben dann checken
171                 // auf error seite verweisen und message schreiben
172 
173                 $this->user = rex_sql::factory($this->DB);
174 
175                 $this->user->setQuery($this->loginQuery, [':login' => $this->userLogin]);
176                 if ($this->user->getRows() == 1 && self::passwordVerify($this->userPassword, $this->user->getValue($this->passwordColumn), true)) {
177                     $ok = true;
178                     self::regenerateSessionId();
179                     $this->setSessionVar('UID', $this->user->getValue($this->idColumn));
180                 } else {
181                     $this->message = rex_i18n::msg('login_error');
182                 }
183             } elseif ($this->getSessionVar('UID') != '') {
184                 // wenn kein login und kein logout dann nach sessiontime checken
185                 // message schreiben und falls falsch auf error verweisen
186 
187                 $this->user = rex_sql::factory($this->DB);
188 
189                 $this->user->setQuery($this->userQuery, [':id' => $this->getSessionVar('UID')]);
190                 if ($this->user->getRows() == 1) {
191                     if (($this->getSessionVar('STAMP') + $this->sessionDuration) > time()) {
192                         $ok = true;
193                         $this->setSessionVar('UID', $this->user->getValue($this->idColumn));
194                     } else {
195                         $this->message = rex_i18n::msg('login_session_expired');
196 
197                         rex_csrf_token::removeAll();
198                     }
199                 } else {
200                     $this->message = rex_i18n::msg('login_user_not_found');
201                 }
202             }
203         } else {
204             $this->message = rex_i18n::msg('login_logged_out');
205 
206             rex_csrf_token::removeAll();
207         }
208 
209         if ($ok) {
210             // wenn alles ok dann REX[UID][system_id] schreiben
211             $this->setSessionVar('STAMP', time());
212 
213             // each code-path which set $ok=true, must also set a UID
214             $sessUid = $this->getSessionVar('UID');
215             if (empty($sessUid)) {
216                 throw new rex_exception('Login considered successfull but no UID found');
217             }
218         } else {
219             // wenn nicht, dann UID loeschen und error seite
220             $this->setSessionVar('STAMP', '');
221             $this->setSessionVar('UID', '');
222         }
223 
224         if ($ok) {
225             $this->loginStatus = 1;
226         } else {
227             $this->loginStatus = -1;
228         }
229 
230         return $ok;
231     }

261     /**
262      * Gibt den Wert einer Session-Variable zurück.
263      */
264     public function getSessionVar($varname, $default = '')
265     {
266         static $sessChecked = false;
267         // validate session-id - once per request - to prevent fixation
268         if (!$sessChecked) {
269             $rexSessId = !empty($_SESSION['REX_SESSID']) ? $_SESSION['REX_SESSID'] : '';
270 
271             if (!empty($rexSessId) && $rexSessId !== session_id()) {
272                 // clear redaxo related session properties on a possible attack
273                 $_SESSION[rex::getProperty('instname')][$this->systemId] = [];
274             }
275             $sessChecked = true;
276         }
277 
278         if (isset($_SESSION[rex::getProperty('instname')][$this->systemId][$varname])) {
279             return $_SESSION[rex::getProperty('instname')][$this->systemId][$varname];
280         }
281 
282         return $default;
283     }

285     /*
286      * refresh session on permission elevation for security reasons
287      */
288     protected static function regenerateSessionId()
289     {
290         if ('' != session_id()) {
291             session_regenerate_id(true);
292 
293             $cookieParams = static::getCookieParams();
294             if ($cookieParams['samesite']) {
295                 self::rewriteSessionCookie($cookieParams['samesite']);
296             }
297 
298             rex_csrf_token::removeAll();
299         }
300 
301         // session-id is shared between frontend/backend or even redaxo instances per server because it's the same http session
302         $_SESSION['REX_SESSID'] = session_id();
303     }

305     /**
306      * starts a http-session if not already started.
307      */
308     public static function startSession()
309     {
310         if (session_id() == '') {
311             $cookieParams = static::getCookieParams();
312 
313             session_set_cookie_params(
314                 $cookieParams['lifetime'],
315                 $cookieParams['path'],
316                 $cookieParams['domain'],
317                 $cookieParams['secure'],
318                 $cookieParams['httponly']
319             );
320 
321             if (!@session_start()) {
322                 $error = error_get_last();
323                 if ($error) {
324                     rex_error_handler::handleError($error['type'], $error['message'], $error['file'], $error['line']);
325                 } else {
326                     throw new rex_exception('Unable to start session!');
327                 }
328             }
329 
330             if ($cookieParams['samesite']) {
331                 self::rewriteSessionCookie($cookieParams['samesite']);
332             }
333         }
334     }

336     /**
337      * Einstellen der Cookie Paramter bevor die session gestartet wird.
338      *
339      * @return array
340      */
341     private static function getCookieParams()
342     {
343         $cookieParams = session_get_cookie_params();
344 
345         $key = rex::isBackend() ? 'backend' : 'frontend';
346         $sessionConfig = rex::getProperty('session');
347 
348         foreach ($sessionConfig[$key]['cookie'] as $name => $value) {
349             if ($value !== null) {
350                 $cookieParams[$name] = $value;
351             }
352         }
353 
354         return $cookieParams;
355     }

357     /**
358      * php does not natively support SameSite for cookies yet,
359      * rewrite the session cookie manually.
360      *
361      * see https://wiki.php.net/rfc/same-site-cookie
362      *
363      * @param "Strict"|"Lax" $sameSite
364      */
365     private static function rewriteSessionCookie($sameSite)
366     {
367         $cookiesHeaders = [];
368 
369         // since header_remove() will remove all sent cookies, we need to collect all of them,
370         // rewrite only the session cookie and send all cookies again.
371         $cookieHeadersPrefix = 'Set-Cookie: ';
372         $sessionCookiePrefix = 'Set-Cookie: '. session_name() .'=';
373         foreach (headers_list() as $rawHeader) {
374             // rewrite the session cookie
375             if (substr($rawHeader, 0, strlen($sessionCookiePrefix)) === $sessionCookiePrefix) {
376                 $rawHeader .= '; SameSite='. $sameSite;
377             }
378             // collect all cookies
379             if (substr($rawHeader, 0, strlen($cookieHeadersPrefix)) === $cookieHeadersPrefix) {
380                 $cookiesHeaders[] = $rawHeader;
381             }
382         }
383 
384         // remove all cookies
385         header_remove('Set-Cookie');
386 
387         // re-add all (inl. the rewritten session cookie)
388         foreach ($cookiesHeaders as $rawHeader) {
389             header($rawHeader);
390         }
391     }
