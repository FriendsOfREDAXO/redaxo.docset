153     /**
154      * Prüft die mit setLogin() und setPassword() gesetzten Werte
155      * anhand des LoginQueries/UserQueries und gibt den Status zurück.
156      *
157      * Gibt true zurück bei erfolg, sonst false
158      */
159     public function checkLogin()
160     {
161         // wenn logout dann header schreiben und auf error seite verweisen
162         // message schreiben
163 
164         $ok = false;
165 
166         if (!$this->logout) {
167             // LoginStatus: 0 = noch checken, 1 = ok, -1 = not ok
168 
169             // gecachte ausgabe erlaubt ? checkLogin schonmal ausgeführt ?
170             if ($this->cache && $this->loginStatus != 0) {
171                 return $this->loginStatus > 0;
172             }
173 
174             if ($this->userLogin != '') {
175                 // wenn login daten eingegeben dann checken
176                 // auf error seite verweisen und message schreiben
177 
178                 $this->user = rex_sql::factory($this->DB);
179 
180                 $this->user->setQuery($this->loginQuery, [':login' => $this->userLogin]);
181                 if ($this->user->getRows() == 1 && self::passwordVerify($this->userPassword, $this->user->getValue($this->passwordColumn), true)) {
182                     $ok = true;
183                     self::regenerateSessionId();
184                     $this->setSessionVar('UID', $this->user->getValue($this->idColumn));
185                 } else {
186                     $this->message = rex_i18n::msg('login_error');
187                 }
188             } elseif ($this->getSessionVar('UID') != '') {
189                 // wenn kein login und kein logout dann nach sessiontime checken
190                 // message schreiben und falls falsch auf error verweisen
191 
192                 $this->user = rex_sql::factory($this->DB);
193 
194                 $this->user->setQuery($this->userQuery, [':id' => $this->getSessionVar('UID')]);
195                 if ($this->user->getRows() == 1) {
196                     if (($this->getSessionVar('STAMP') + $this->sessionDuration) > time()) {
197                         $ok = true;
198                         $this->setSessionVar('UID', $this->user->getValue($this->idColumn));
199 
200                         if ($impersonator = $this->getSessionVar('impersonator')) {
201                             $this->impersonator = rex_sql::factory($this->DB);
202                             $this->impersonator->setQuery($this->userQuery, [':id' => $impersonator]);
203 
204                             if (!$this->impersonator->getRows()) {
205                                 $ok = false;
206                                 $this->message = rex_i18n::msg('login_user_not_found');
207                             }
208                         }
209                     } else {
210                         $this->message = rex_i18n::msg('login_session_expired');
211 
212                         rex_csrf_token::removeAll();
213                     }
214                 } else {
215                     $this->message = rex_i18n::msg('login_user_not_found');
216                 }
217             }
218         } else {
219             $this->message = rex_i18n::msg('login_logged_out');
220 
221             rex_csrf_token::removeAll();
222         }
223 
224         if ($ok) {
225             // wenn alles ok dann REX[UID][system_id] schreiben
226             $this->setSessionVar('STAMP', time());
227 
228             // each code-path which set $ok=true, must also set a UID
229             $sessUid = $this->getSessionVar('UID');
230             if (empty($sessUid)) {
231                 throw new rex_exception('Login considered successfull but no UID found');
232             }
233         } else {
234             // wenn nicht, dann UID loeschen und error seite
235             $this->setSessionVar('STAMP', '');
236             $this->setSessionVar('UID', '');
237             $this->setSessionVar('impersonator', null);
238         }
239 
240         if ($ok) {
241             $this->loginStatus = 1;
242         } else {
243             $this->loginStatus = -1;
244         }
245 
246         return $ok;
247     }

249     public function impersonate($id)
250     {
251         if (!$this->user) {
252             throw new RuntimeException('Can not impersonate a user without valid user session.');
253         }
254         if ($this->user->getValue($this->idColumn) == $id) {
255             throw new RuntimeException('Can not impersonate the current user.');
256         }
257 
258         $user = rex_sql::factory($this->DB);
259         $user->setQuery($this->userQuery, [':id' => $id]);
260 
261         if (!$user->getRows()) {
262             throw new RuntimeException(sprintf('User with id "%d" not found.', $id));
263         }
264 
265         $this->impersonator = $this->user;
266         $this->user = $user;
267 
268         $this->setSessionVar('UID', $id);
269         $this->setSessionVar('impersonator', $this->impersonator->getValue($this->idColumn));
270     }
271 
272     public function depersonate()
273     {
274         if (!$this->impersonator) {
275             throw new RuntimeException('There is no current impersonator.');
276         }
277 
278         $this->user = $this->impersonator;
279         $this->impersonator = null;
280 
281         $this->setSessionVar('UID', $this->user->getValue($this->idColumn));
282         $this->setSessionVar('impersonator', null);
283     }

321     /**
322      * Gibt den Wert einer Session-Variable zurück.
323      */
324     public function getSessionVar($varname, $default = '')
325     {
326         static $sessChecked = false;
327         // validate session-id - once per request - to prevent fixation
328         if (!$sessChecked) {
329             $rexSessId = !empty($_SESSION['REX_SESSID']) ? $_SESSION['REX_SESSID'] : '';
330 
331             if (!empty($rexSessId) && $rexSessId !== session_id()) {
332                 // clear redaxo related session properties on a possible attack
333                 $_SESSION[rex::getProperty('instname')][$this->systemId] = [];
334             }
335             $sessChecked = true;
336         }
337 
338         if (isset($_SESSION[rex::getProperty('instname')][$this->systemId][$varname])) {
339             return $_SESSION[rex::getProperty('instname')][$this->systemId][$varname];
340         }
341 
342         return $default;
343     }

345     /*
346      * refresh session on permission elevation for security reasons
347      */
348     protected static function regenerateSessionId()
349     {
350         if ('' != session_id()) {
351             session_regenerate_id(true);
352 
353             $cookieParams = static::getCookieParams();
354             if ($cookieParams['samesite']) {
355                 self::rewriteSessionCookie($cookieParams['samesite']);
356             }
357 
358             rex_csrf_token::removeAll();
359         }
360 
361         // session-id is shared between frontend/backend or even redaxo instances per server because it's the same http session
362         $_SESSION['REX_SESSID'] = session_id();
363     }

365     /**
366      * starts a http-session if not already started.
367      */
368     public static function startSession()
369     {
370         if (session_id() == '') {
371             $cookieParams = static::getCookieParams();
372 
373             session_set_cookie_params(
374                 $cookieParams['lifetime'],
375                 $cookieParams['path'],
376                 $cookieParams['domain'],
377                 $cookieParams['secure'],
378                 $cookieParams['httponly']
379             );
380 
381             if (!@session_start()) {
382                 $error = error_get_last();
383                 if ($error) {
384                     rex_error_handler::handleError($error['type'], $error['message'], $error['file'], $error['line']);
385                 } else {
386                     throw new rex_exception('Unable to start session!');
387                 }
388             }
389 
390             if ($cookieParams['samesite']) {
391                 self::rewriteSessionCookie($cookieParams['samesite']);
392             }
393         }
394     }

396     /**
397      * Einstellen der Cookie Paramter bevor die session gestartet wird.
398      *
399      * @return array
400      */
401     private static function getCookieParams()
402     {
403         $cookieParams = session_get_cookie_params();
404 
405         $key = rex::isBackend() ? 'backend' : 'frontend';
406         $sessionConfig = rex::getProperty('session');
407 
408         foreach ($sessionConfig[$key]['cookie'] as $name => $value) {
409             if ($value !== null) {
410                 $cookieParams[$name] = $value;
411             }
412         }
413 
414         return $cookieParams;
415     }

417     /**
418      * php does not natively support SameSite for cookies yet,
419      * rewrite the session cookie manually.
420      *
421      * see https://wiki.php.net/rfc/same-site-cookie
422      *
423      * @param "Strict"|"Lax" $sameSite
424      */
425     private static function rewriteSessionCookie($sameSite)
426     {
427         $cookiesHeaders = [];
428 
429         // since header_remove() will remove all sent cookies, we need to collect all of them,
430         // rewrite only the session cookie and send all cookies again.
431         $cookieHeadersPrefix = 'Set-Cookie: ';
432         $sessionCookiePrefix = 'Set-Cookie: '. session_name() .'=';
433         foreach (headers_list() as $rawHeader) {
434             // rewrite the session cookie
435             if (substr($rawHeader, 0, strlen($sessionCookiePrefix)) === $sessionCookiePrefix) {
436                 $rawHeader .= '; SameSite='. $sameSite;
437             }
438             // collect all cookies
439             if (substr($rawHeader, 0, strlen($cookieHeadersPrefix)) === $cookieHeadersPrefix) {
440                 $cookiesHeaders[] = $rawHeader;
441             }
442         }
443 
444         // remove all cookies
445         header_remove('Set-Cookie');
446 
447         // re-add all (inl. the rewritten session cookie)
448         foreach ($cookiesHeaders as $rawHeader) {
449             header($rawHeader);
450         }
451     }
