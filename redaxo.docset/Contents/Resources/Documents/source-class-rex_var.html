 17     private static $context = null;
 18     private static $contextData = null;

 20     private $args = [];

 24     /**
 25      * Parses all REX_VARs in the given content.
 26      *
 27      * @param string $content     Content
 28      * @param int    $env         Environment
 29      * @param string $context     Context
 30      * @param mixed  $contextData Context data
 31      *
 32      * @return string
 33      */
 34     public static function parse($content, $env = null, $context = null, $contextData = null)
 35     {
 36         $env = (int) $env;
 37 
 38         if (($env & self::ENV_INPUT) != self::ENV_INPUT) {
 39             $env = $env | self::ENV_OUTPUT;
 40         }
 41 
 42         self::$env = $env;
 43         self::$context = $context;
 44         self::$contextData = $contextData;
 45 
 46         self::$variableIndex = 0;
 47 
 48         $tokens = token_get_all($content);
 49         $countTokens = count($tokens);
 50         $content = '';
 51         for ($i = 0; $i < $countTokens; ++$i) {
 52             $token = $tokens[$i];
 53             if (is_string($token)) {
 54                 $content .= $token;
 55                 continue;
 56             }
 57 
 58             if (!in_array($token[0], [T_INLINE_HTML, T_CONSTANT_ENCAPSED_STRING, T_STRING, T_START_HEREDOC])) {
 59                 $content .= $token[1];
 60                 continue;
 61             }
 62 
 63             $add = $token[1];
 64             switch ($token[0]) {
 65                 case T_INLINE_HTML:
 66                     $format = '<?= %s@@@INLINE_HTML_REPLACEMENT_END@@@';
 67                     $add = self::replaceVars($add, $format);
 68                     $add = preg_replace_callback('/@@@INLINE_HTML_REPLACEMENT_END@@@(\r?\n?)/', function (array $match) {
 69                         return $match[1]
 70                             ? ', "'.addcslashes($match[1], "\r\n").'" ?>'.$match[1]
 71                             : ' ?>';
 72                     }, $add);
 73                     break;
 74 
 75                 case T_CONSTANT_ENCAPSED_STRING:
 76                     $format = $token[1][0] == '"' ? '" . %s . "' : "' . %s . '";
 77                     $add = self::replaceVars($add, $format, false, $token[1][0]);
 78 
 79                     $start = substr($add, 0, 5);
 80                     $end = substr($add, -5);
 81                     if ($start == '"" . ' || $start == "'' . ") {
 82                         $add = substr($add, 5);
 83                     }
 84                     if ($end == ' . ""' || $end == " . ''") {
 85                         $add = substr($add, 0, -5);
 86                     }
 87                     break;
 88 
 89                 case T_STRING:
 90                     while (isset($tokens[++$i])
 91                         && (is_string($tokens[$i]) && in_array($tokens[$i], ['=', '[', ']'])
 92                             || in_array($tokens[$i][0], [T_WHITESPACE, T_STRING, T_CONSTANT_ENCAPSED_STRING, T_LNUMBER, T_ISSET]))
 93                     ) {
 94                         $add .= is_string($tokens[$i]) ? $tokens[$i] : $tokens[$i][1];
 95                     }
 96                     --$i;
 97                     $add = self::replaceVars($add);
 98                     break;
 99 
100                 case T_START_HEREDOC:
101                     while (isset($tokens[++$i]) && (is_string($tokens[$i]) || $tokens[$i][0] != T_END_HEREDOC)) {
102                         $add .= is_string($tokens[$i]) ? $tokens[$i] : $tokens[$i][1];
103                     }
104                     --$i;
105                     if (preg_match("/'(.*)'/", $token[1], $match)) { // nowdoc
106                         $format = "\n" . $match[1] . "\n. %s . <<<'" . $match[1] . "'\n";
107                         $add = self::replaceVars($add, $format);
108                     } else { // heredoc
109                         $add = self::replaceVars($add, '{%s}', true);
110                     }
111                     break;
112             }
113 
114             $content .= $add;
115         }
116         return $content;
117     }

119     /**
120      * Returns a rex_var object for the given var name.
121      *
122      * @param string $var
123      *
124      * @return self
125      */
126     private static function getVar($var)
127     {
128         if (!isset(self::$vars[$var])) {
129             $class = 'rex_var_' . strtolower(substr($var, 4));
130             if (!class_exists($class) || !is_subclass_of($class, __CLASS__)) {
131                 return false;
132             }
133             self::$vars[$var] = $class;
134         }
135         $class = self::$vars[$var];
136         return new $class();
137     }

139     /**
140      * Replaces the REX_VARs.
141      *
142      * @param string $content
143      * @param string $format
144      * @param bool   $useVariables
145      * @param string $stripslashes
146      *
147      * @return mixed|string
148      */
149     private static function replaceVars($content, $format = '%s', $useVariables = false, $stripslashes = null)
150     {
151         $matches = self::getMatches($content);
152 
153         if (empty($matches)) {
154             return $content;
155         }
156 
157         $iterator = new AppendIterator();
158         $iterator->append(new ArrayIterator($matches));
159         $variables = [];
160         foreach ($iterator as $match) {
161             $var = self::getVar($match[1]);
162             $replaced = false;
163 
164             if ($var !== false) {
165                 $args = str_replace(['\[', '\]'], ['@@@OPEN_BRACKET@@@', '@@@CLOSE_BRACKET@@@'], $match[2]);
166                 if ($stripslashes) {
167                     $args = str_replace(['\\' . $stripslashes, '\\' . $stripslashes], $stripslashes, $args);
168                 }
169                 $var->setArgs($args);
170                 if (($output = $var->getGlobalArgsOutput()) !== false) {
171                     $output .= str_repeat("\n", max(0, substr_count($match[0], "\n") - substr_count($output, "\n") - substr_count($format, "\n")));
172                     if ($useVariables) {
173                         $replace = '$__rex_var_content_' . ++self::$variableIndex;
174                         $variables[] = $replace . ' = ' . $output;
175                     } else {
176                         $replace = $output;
177                     }
178                     $content = str_replace($match[0], sprintf($format, $replace), $content);
179                     $replaced = true;
180                 }
181             }
182 
183             if (!$replaced && $matches = self::getMatches($match[2])) {
184                 $iterator->append(new ArrayIterator($matches));
185             }
186         }
187 
188         if ($useVariables && !empty($variables)) {
189             $content = 'rex_var::nothing(' . implode(', ', $variables) . ') . ' . $content;
190         }
191         return $content;
192     }

194     /**
195      * Returns the REX_VAR matches.
196      *
197      * @param string $content
198      *
199      * @return array
200      */
201     private static function getMatches($content)
202     {
203         preg_match_all('/(REX_[A-Z_]+)\[((?:[^\[\]]|\\\\[\[\]]|(?R))*)(?<!\\\\)\]/s', $content, $matches, PREG_SET_ORDER);
204         return $matches;
205     }

217     /**
218      * Checks whether the given arguments exists.
219      *
220      * @param string $key
221      * @param bool   $defaultArg
222      *
223      * @return bool

225     protected function hasArg($key, $defaultArg = false)
226     {
227         return isset($this->args[$key]) || $defaultArg && isset($this->args[0]);
228     }

230     /**
231      * Returns the argument.
232      *
233      * @param string      $key
234      * @param null|string $default
235      * @param bool        $defaultArg
236      *
237      * @return null|string
238      */
239     protected function getArg($key, $default = null, $defaultArg = false)
240     {
241         if (!$this->hasArg($key, $defaultArg)) {
242             return $default;
243         }
244         return isset($this->args[$key]) ? $this->args[$key] : $this->args[0];
245     }

247     /**
248      * Returns the (recursive) parsed argument.
249      *
250      * @param string      $key
251      * @param null|string $default
252      * @param bool        $defaultArg
253      *
254      * @return int|null|string
255      */
256     protected function getParsedArg($key, $default = null, $defaultArg = false)
257     {
258         if (!$this->hasArg($key, $defaultArg)) {
259             return $default;
260         }
261         $arg = isset($this->args[$key]) ? $this->args[$key] : $this->args[0];
262         $begin = '<<<addslashes>>>';
263         $end = '<<</addslashes>>>';
264         $arg = $begin . self::replaceVars($arg, $end . "' . %s . '" . $begin) . $end;
265         $arg = preg_replace_callback("@$begin(.*)$end@Us", function ($match) {
266             return addcslashes($match[1], "\'");
267         }, $arg);
268         $arg = str_replace(['@@@OPEN_BRACKET@@@', '@@@CLOSE_BRACKET@@@'], ['[', ']'], $arg);
269         return is_numeric($arg) ? $arg : "'$arg'";
270     }

311     /**
312      * Quotes the string for php context.
313      *
314      * @param string $string
315      *
316      * @return string
317      */
318     protected static function quote($string)
319     {
320         $string = addcslashes($string, "\\'");
321         $string = preg_replace('/\v+/', '\' . "$0" . \'', $string);
322         $string = addcslashes($string, "\r\n");
323         return "'" . $string . "'";
324     }

326     /**
327      * Returns the output in consideration of the global args.
328      *
329      * @return bool|string
330      */
331     private function getGlobalArgsOutput()
332     {
333         if (($content = $this->getOutput()) === false) {
334             return false;
335         }
336 
337         if ($this->hasArg('callback')) {
338             $args = ["'subject' => " . $content];
339             foreach ($this->args as $key => $value) {
340                 $args[] = "'$key' => " . $this->getParsedArg($key);
341             }
342             $args = '[' . implode(', ', $args) . ']';
343             return 'call_user_func(' . $this->getParsedArg('callback') . ', ' . $args . ')';
344         }
345 
346         $prefix = $this->hasArg('prefix') ? $this->getParsedArg('prefix') . ' . ' : '';
347         $suffix = $this->hasArg('suffix') ? ' . ' . $this->getParsedArg('suffix') : '';
348         $instead = $this->hasArg('instead');
349         $ifempty = $this->hasArg('ifempty');
350         if ($prefix || $suffix || $instead || $ifempty) {
351             if ($instead) {
352                 $if = $content;
353                 $then = $this->getParsedArg('instead');
354             } else {
355                 $if = '$__rex_var_content_' . ++self::$variableIndex . ' = ' . $content;
356                 $then = '$__rex_var_content_' . self::$variableIndex;
357             }
358             if ($ifempty) {
359                 return $prefix . '((' . $if . ') ? ' . $then . ' : ' . $this->getParsedArg('ifempty') . ')' . $suffix;
360             }
361             return '((' . $if . ') ? ' . $prefix . $then . $suffix . " : '')";
362         }
363         return $content;
364     }
