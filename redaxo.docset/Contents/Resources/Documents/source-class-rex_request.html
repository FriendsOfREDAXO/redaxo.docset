 66     /**
 67      * Returns the variable $varname of $_SESSION and casts the value.
 68      *
 69      * @param string $varname Variable name
 70      * @param string $vartype Variable type
 71      * @param mixed  $default Default value
 72      *
 73      * @throws rex_exception
 74      *
 75      * @return mixed
 76      */
 77     public static function session($varname, $vartype = '', $default = '')
 78     {
 79         if (PHP_SESSION_ACTIVE != session_status()) {
 80             throw new rex_exception('Session not started, call rex_login::startSession() before!');
 81         }
 82 
 83         if (isset($_SESSION[rex::getProperty('instname')][$varname])) {
 84             return rex_type::cast($_SESSION[rex::getProperty('instname')][$varname], $vartype);
 85         }
 86 
 87         if ($default === '') {
 88             return rex_type::cast($default, $vartype);
 89         }
 90         return $default;
 91     }

168     /**
169      * Searches the value $needle in array $haystack and returns the casted value.
170      *
171      * @param array      $haystack Array
172      * @param string|int $needle   Value to search
173      * @param string     $vartype  Variable type
174      * @param mixed      $default  Default value
175      *
176      * @throws InvalidArgumentException
177      *
178      * @return mixed
179      */
180     private static function arrayKeyCast(array $haystack, $needle, $vartype, $default = '')
181     {
182         if (!is_scalar($needle)) {
183             throw new InvalidArgumentException('Scalar expected for $needle in arrayKeyCast()!');
184         }
185 
186         if (array_key_exists($needle, $haystack)) {
187             return rex_type::cast($haystack[$needle], $vartype);
188         }
189 
190         if ($default === '') {
191             return rex_type::cast($default, $vartype);
192         }
193         return $default;
194     }

196     /**
197      * Returns the HTTP method of the current request.
198      *
199      * @return string HTTP method in lowercase (head,get,post,put,delete)
200      */
201     public static function requestMethod()
202     {
203         return isset($_SERVER['REQUEST_METHOD']) ? strtolower($_SERVER['REQUEST_METHOD']) : 'get';
204     }

206     /**
207      * Returns true if the request is a XMLHttpRequest.
208      *
209      * This only works if your javaScript library sets an X-Requested-With HTTP header.
210      * This is the case with Prototype, Mootools, jQuery, and perhaps others.
211      *
212      * Inspired by a method of the symfony framework.
213      *
214      * @return bool true if the request is an XMLHttpRequest, false otherwise
215      */
216     public static function isXmlHttpRequest()
217     {
218         return isset($_SERVER['HTTP_X_REQUESTED_WITH']) && $_SERVER['HTTP_X_REQUESTED_WITH'] == 'XMLHttpRequest';
219     }
220 
221     /**
222      * Returns true if the request is a PJAX-Request.
223      *
224      * @see http://pjax.heroku.com/
225      */
226     public static function isPJAXRequest()
227     {
228         return isset($_SERVER['HTTP_X_PJAX']) && $_SERVER['HTTP_X_PJAX'] == 'true';
229     }
230 
231     /**
232      * Returns true when the current request is a PJAX-Request and the requested container matches the given $containerId.
233      *
234      * @param string $containerId
235      *
236      * @return bool
237      */
238     public static function isPJAXContainer($containerId)
239     {
240         if (!self::isPJAXRequest()) {
241             return false;
242         }
243 
244         return isset($_SERVER['HTTP_X_PJAX_CONTAINER']) && $_SERVER['HTTP_X_PJAX_CONTAINER'] == $containerId;
245     }

248      * Returns whether the current request is served via https/ssl.
249      *
250      * @return bool true when https/ssl, otherwise false.
251      */
252     public static function isHttps()
253     {
254         return !empty($_SERVER['HTTPS']) && 'off' !== strtolower($_SERVER['HTTPS']);
