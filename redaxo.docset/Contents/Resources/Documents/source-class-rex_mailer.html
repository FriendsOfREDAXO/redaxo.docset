11 use PHPMailer\PHPMailer\PHPMailer;
12 
13 class rex_mailer extends PHPMailer
14 {
15     private $log;
16 
17     public function __construct($exceptions = false)
18     {
19         $addon = rex_addon::get('phpmailer');
20         $this->XMailer = 'REXMailer';
21         $this->From = $addon->getConfig('from');
22         $this->FromName = $addon->getConfig('fromname');
23         $this->ConfirmReadingTo = $addon->getConfig('confirmto');
24         $this->Mailer = $addon->getConfig('mailer');
25         $this->Host = $addon->getConfig('host');
26         $this->Port = $addon->getConfig('port');
27         $this->CharSet = $addon->getConfig('charset');
28         $this->WordWrap = $addon->getConfig('wordwrap');
29         $this->Encoding = $addon->getConfig('encoding');
30         if ($addon->getConfig('priority') == 0) {
31             $this->Priority = null;
32         } else {
33             $this->Priority = $addon->getConfig('priority');
34         }
35         $this->SMTPDebug = $addon->getConfig('smtp_debug');
36         $this->SMTPSecure = $addon->getConfig('smtpsecure');
37         $this->SMTPAuth = $addon->getConfig('smtpauth');
38         $this->Username = $addon->getConfig('username');
39         $this->Password = $addon->getConfig('password');
40 
41         if ($bcc = $addon->getConfig('bcc')) {
42             $this->addBCC($bcc);
43         }
44 
45         $this->log = $addon->getConfig('log');
46 
47         parent::__construct($exceptions);
48     }
49 
50     public function send()
51     {
52         if ($this->log) {
53             $this->log();
54         }
55         return parent::send();
56     }
57 
58     /*
59      * @param boolean $status
60      */
61     public function setLog($status)
62     {
63         $this->log = $status;
64     }
65 
66     private function log()
67     {
68         $content = '<!-- '.PHP_EOL.date('d.m.Y H:i:s').PHP_EOL;
69         $content .= 'From : '.$this->From.PHP_EOL;
70         $content .= 'To : '.implode(', ', array_column($this->getToAddresses(), 0)).PHP_EOL;
71         $content .= 'Subject : '.$this->Subject.PHP_EOL;
72         $content .= ' -->'.PHP_EOL;
73         $content .= $this->Body;
74 
75         $dir = self::logFolder().'/'.date('Y').'/'.date('m');
76 
77         $count = 1;
78         $logFile = $dir.'/'.date('Y-m-d_H_i_s').'.html';
79         while (file_exists($logFile)) {
80             $logFile = $dir.'/'.date('Y-m-d_H_i_s').'_'.(++$count).'.html';
81         }
82 
83         rex_file::put($logFile, $content);
84     }
85 
86     public static function logFolder()
87     {
88         return rex_path::addonData('phpmailer', 'mail_log');
89     }
90 }
