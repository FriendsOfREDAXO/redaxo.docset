 3 /**
 4  * Class to stop the script time.
 5  *
 6  * @author gharlan
 7  *
 8  * @package redaxo\core
 9  */
10 class rex_timer
11 {
12     const SEC = 1;
13     const MILLISEC = 1000;
14     const MICROSEC = 1000000;
15 
16     private $start;
17 
18     /**
19      * Constructor.
20      *
21      * @param float $start Start time
22      */
23     public function __construct($start = null)
24     {
25         if ($start) {
26             $this->start = $start;
27         } else {
28             $this->reset();
29         }
30     }
31 
32     /**
33      * Resets the timer.
34      */
35     public function reset()
36     {
37         $this->start = microtime(true);
38     }
39 
40     /**
41      * Returns the time difference.
42      *
43      * @param int $precision Factor which will be multiplied, for convertion into different units (e.g. 1000 for milli,...)
44      *
45      * @return float Time difference
46      */
47     public function getDelta($precision = self::MILLISEC)
48     {
49         return (microtime(true) - $this->start) * $precision;
50     }
51 
52     /**
53      * Returns the formatted time difference.
54      *
55      * @param int $precision Factor which will be multiplied, for convertion into different units (e.g. 1000 for milli,...)
56      * @param int $decimals  Number of decimals points
57      *
58      * @return string Formatted time difference
59      */
60     public function getFormattedDelta($precision = self::MILLISEC, $decimals = 3)
61     {
62         $time = $this->getDelta($precision);
63         return rex_formatter::number($time, [$decimals]);
64     }
65 }
