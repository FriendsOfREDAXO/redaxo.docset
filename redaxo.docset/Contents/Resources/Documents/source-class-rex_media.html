 45     /**
 46      * @param string $name
 47      *
 48      * @return null|static
 49      */
 50     public static function get($name)
 51     {
 52         if (!$name) {
 53             return null;
 54         }
 55 
 56         return static::getInstance($name, function ($name) {
 57             $media_path = rex_path::addonCache('mediapool', $name . '.media');
 58             if (!file_exists($media_path)) {
 59                 rex_media_cache::generate($name);
 60             }
 61 
 62             if (file_exists($media_path)) {
 63                 $cache = rex_file::getCache($media_path);
 64                 $aliasMap = [
 65                     'filename' => 'name',
 66                     'filetype' => 'type',
 67                     'filesize' => 'size',
 68                 ];
 69 
 70                 $media = new static();
 71                 foreach ($cache as $key => $value) {
 72                     if (isset($aliasMap[$key])) {
 73                         $var_name = $aliasMap[$key];
 74                     } else {
 75                         $var_name = $key;
 76                     }
 77 
 78                     $media->$var_name = $value;
 79                 }
 80                 $media->category = null;
 81 
 82                 return $media;
 83             }
 84 
 85             return null;
 86         });
 87     }

232     /**
233      * @param array $params
234      *
235      * @return string
236      */
237     public function toImage(array $params = [])
238     {
239         if (!$this->isImage()) {
240             return '';
241         }
242 
243         $filename = rex_url::media($this->getFileName());
244         $title = $this->getTitle();
245 
246         if (!isset($params['alt'])) {
247             if ($title != '') {
248                 $params['alt'] = htmlspecialchars($title);
249             }
250         }
251 
252         if (!isset($params['title'])) {
253             if ($title != '') {
254                 $params['title'] = htmlspecialchars($title);
255             }
256         }
257 
258         rex_extension::registerPoint(new rex_extension_point('MEDIA_TOIMAGE', '', ['filename' => &$filename, 'params' => &$params]));
259 
260         $additional = '';
261         foreach ($params as $name => $value) {
262             $additional .= ' ' . $name . '="' . $value . '"';
263         }
264 
265         return sprintf('<img src="%s"%s />', $filename, $additional);
266     }
