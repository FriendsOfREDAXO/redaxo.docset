 33     /**
 34      * @param int $id
 35      *
 36      * @return self
 37      */
 38     public static function get($id)
 39     {
 40         $id = (int) $id;
 41 
 42         if (0 >= $id) {
 43             return null;
 44         }
 45 
 46         return self::getInstance($id, function ($id) {
 47             $cat_path = rex_path::addonCache('mediapool', $id . '.mcat');
 48             if (!file_exists($cat_path)) {
 49                 rex_media_cache::generateCategory($id);
 50             }
 51 
 52             if (file_exists($cat_path)) {
 53                 $cache = rex_file::getCache($cat_path);
 54 
 55                 $cat = new self();
 56 
 57                 $cat->id = $cache['id'];
 58                 $cat->parent_id = $cache['parent_id'];
 59 
 60                 $cat->name = $cache['name'];
 61                 $cat->path = $cache['path'];
 62 
 63                 $cat->createdate = $cache['createdate'];
 64                 $cat->updatedate = $cache['updatedate'];
 65 
 66                 $cat->createuser = $cache['createuser'];
 67                 $cat->updateuser = $cache['updateuser'];
 68 
 69                 return $cat;
 70             }
 71 
 72             return null;
 73         });
 74     }
 75 
 76     /**
 77      * @return self[]
 78      */
 79     public static function getRootCategories()
 80     {
 81         return self::getChildCategories(0);
 82     }
 83 
 84     /**
 85      * @param int $parentId
 86      *
 87      * @return self[]
 88      */
 89     protected static function getChildCategories($parentId)
 90     {
 91         $parentId = (int) $parentId;
 92         // for $parentId=0 root categories will be returned, so abort here for $parentId<0 only
 93         if (0 > $parentId) {
 94             return [];
 95         }
 96 
 97         return self::getInstanceList([$parentId, 'children'], 'self::get', function ($parentId) {
 98             $catlist_path = rex_path::addonCache('mediapool', $parentId . '.mclist');
 99             if (!file_exists($catlist_path)) {
100                 rex_media_cache::generateCategoryList($parentId);
101             }
102             return rex_file::getCache($catlist_path);
103         });
104     }

130     /**
131      * Returns the path ids of the category as an array.
132      *
133      * @return int[]
134      */
135     public function getPathAsArray()
136     {
137         $p = explode('|', $this->path);
138         foreach ($p as $k => $v) {
139             if ($v == '') {
140                 unset($p[$k]);
141             } else {
142                 $p[$k] = (int) $v;
143             }
144         }
145 
146         return array_values($p);
147     }

197     /**
198      * Get an array of all parentCategories.
199      * Returns an array of rex_media_category objects sorted by $priority.
200      *
201      * @return self[]
202      */
203     public function getParentTree()
204     {
205         $tree = [];
206         if ($this->path) {
207             $explode = explode('|', $this->path);
208             if (is_array($explode)) {
209                 foreach ($explode as $var) {
210                     if ($var != '') {
211                         $tree[] = self::get($var);
212                     }
213                 }
214             }
215         }
216         return $tree;
217     }
218 
219     /**
220      * Checks if $anObj is in the parent tree of the object.
221      *
222      * @param self $anObj
223      *
224      * @return bool
225      */
226     public function inParentTree($anObj)
227     {
228         $tree = $this->getParentTree();
229         foreach ($tree as $treeObj) {
230             if ($treeObj == $anObj) {
231                 return true;
232             }
233         }
234         return false;
235     }
