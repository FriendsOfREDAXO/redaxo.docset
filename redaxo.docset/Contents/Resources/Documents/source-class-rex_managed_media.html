 82     public function asImage()
 83     {
 84         if ($this->asImage) {
 85             return;
 86         }
 87 
 88         $this->asImage = true;
 89 
 90         $this->image = [];
 91         $this->image['src'] = false;
 92 
 93         // if mimetype detected and in imagemap -> change format
 94         if (class_exists('finfo') && $finfo = new finfo(FILEINFO_MIME_TYPE)) {
 95             if ($ftype = @$finfo->file($this->getSourcePath())) {
 96                 if (array_key_exists($ftype, $this->mimetypeMap)) {
 97                     $this->format = $this->mimetypeMap[$ftype];
 98                 }
 99             }
100         }
101 
102         if ($this->format == 'jpg' || $this->format == 'jpeg') {
103             $this->format = 'jpeg';
104             $this->image['src'] = @imagecreatefromjpeg($this->getSourcePath());
105         } elseif ($this->format == 'gif') {
106             $this->image['src'] = @imagecreatefromgif($this->getSourcePath());
107         } elseif ($this->format == 'wbmp') {
108             $this->image['src'] = @imagecreatefromwbmp($this->getSourcePath());
109         } elseif ($this->format == 'webp') {
110             if (function_exists('imagecreatefromwebp')) {
111                 $this->image['src'] = @imagecreatefromwebp($this->getSourcePath());
112                 imagealphablending($this->image['src'], false);
113                 imagesavealpha($this->image['src'], true);
114             }
115         } else {
116             $this->image['src'] = @imagecreatefrompng($this->getSourcePath());
117             if ($this->image['src']) {
118                 imagealphablending($this->image['src'], false);
119                 imagesavealpha($this->image['src'], true);
120                 $this->format = 'png';
121             }
122         }
123 
124         if (!$this->image['src']) {
125             $this->setSourcePath(rex_path::addon('media_manager', 'media/warning.jpg'));
126             $this->asImage();
127         } else {
128             $this->fixOrientation();
129             $this->refreshImageDimensions();
130         }
131     }

133     public function refreshImageDimensions()
134     {
135         // getimagesize does not work for webp with PHP < 7.1
136         if (!$this->asImage && 'webp' === $this->format && PHP_VERSION_ID < 70100) {
137             $this->asImage();
138         }
139 
140         if ($this->asImage) {
141             $this->image['width'] = imagesx($this->image['src']);
142             $this->image['height'] = imagesy($this->image['src']);
143 
144             return;
145         }
146 
147         if ('jpeg' !== $this->format && !in_array($this->format, $this->mimetypeMap)) {
148             return;
149         }
150 
151         $size = getimagesize($this->sourcePath);
152         $this->image['width'] = $size[0];
153         $this->image['height'] = $size[1];
154     }

166     public function sendMedia($sourceCacheFilename, $headerCacheFilename, $save = false)
167     {
168         $src = $this->getSource();
169 
170         $this->prepareHeaders($src);
171 
172         rex_response::cleanOutputBuffers();
173         foreach ($this->header as $t => $c) {
174             header($t . ': ' . $c);
175         }
176         echo $src;
177 
178         if ($save) {
179             $this->saveFiles($src, $sourceCacheFilename, $headerCacheFilename);
180         }
181     }

191     protected function getImageSource()
192     {
193         $addon = rex_addon::get('media_manager');
194 
195         $format = $this->format;
196         $format = 'jpeg' === $format ? 'jpg' : $format;
197 
198         $interlace = $this->getImageProperty('interlace', $addon->getConfig('interlace', ['jpg']));
199         imageinterlace($this->image['src'], in_array($format, $interlace) ? 1 : 0);
200 
201         ob_start();
202         if ($format == 'jpg') {
203             $quality = $this->getImageProperty('jpg_quality', $addon->getConfig('jpg_quality', 85));
204             imagejpeg($this->image['src'], null, $quality);
205         } elseif ($format == 'png') {
206             $compression = $this->getImageProperty('png_compression', $addon->getConfig('png_compression', 5));
207             imagepng($this->image['src'], null, $compression);
208         } elseif ($format == 'gif') {
209             imagegif($this->image['src']);
210         } elseif ($format == 'wbmp') {
211             imagewbmp($this->image['src']);
212         } elseif ($format == 'webp') {
213             $quality = $this->getImageProperty('webp_quality', $addon->getConfig('webp_quality', 85));
214             imagewebp($this->image['src'], null, $quality);
215         }
216         return ob_get_clean();
217     }

301     private function fixOrientation()
302     {
303         if (!function_exists('exif_read_data')) {
304             return;
305         }
306         // exif_read_data() only works on jpg/jpeg/tiff
307         if (!in_array($this->getFormat(), ['jpg', 'jpeg', 'tiff'])) {
308             return;
309         }
310         // suppress warning in case of corrupt/ missing exif data
311         $exif = @exif_read_data($this->getSourcePath());
312 
313         if (!isset($exif['Orientation']) || !in_array($exif['Orientation'], [3, 6, 8])) {
314             return;
315         }
316 
317         switch ($exif['Orientation']) {
318             case 8:
319                 $this->image['src'] = imagerotate($this->image['src'], 90, 0);
320                 break;
321             case 3:
322                 $this->image['src'] = imagerotate($this->image['src'], 180, 0);
323                 break;
324             case 6:
325                 $this->image['src'] = imagerotate($this->image['src'], -90, 0);
326                 break;
327         }
328     }

330     /**
331      * @param string $src Source content
332      */
333     private function prepareHeaders($src)
334     {
335         $this->setHeader('Content-Length', rex_string::size($src));
336 
337         $header = $this->getHeader();
338         if (!isset($header['Content-Type'])) {
339             $content_type = '';
340 
341             if (!$content_type && function_exists('mime_content_type')) {
342                 $content_type = mime_content_type($this->getSourcePath());
343             }
344 
345             if (!$content_type && function_exists('finfo_open')) {
346                 $finfo = finfo_open(FILEINFO_MIME_TYPE);
347                 $content_type = finfo_file($finfo, $this->getSourcePath());
348             }
349 
350             if ($content_type != '') {
351                 $this->setHeader('Content-Type', $content_type);
352             }
353         }
354         if (!isset($header['Content-Disposition'])) {
355             $this->setHeader('Content-Disposition', 'inline; filename="' . $this->getMediaFilename() . '";');
356         }
357         if (!isset($header['Last-Modified'])) {
358             $this->setHeader('Last-Modified', gmdate('D, d M Y H:i:s T'));
359         }
360     }
