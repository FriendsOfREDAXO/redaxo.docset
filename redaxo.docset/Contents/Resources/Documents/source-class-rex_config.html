  3 /**
  4  * Class for handling configurations.
  5  * The configuration is persisted between requests.
  6  *
  7  * @author staabm
  8  *
  9  * @package redaxo\core
 10  */
 11 class rex_config
 12 {
 13     /**
 14      * Flag to indicate if the config was initialized.
 15      *
 16      * @var bool
 17      */
 18     private static $initialized = false;
 19 
 20     /**
 21      * Flag which indicates if database needs an update, because settings have changed.
 22      *
 23      * @var bool
 24      */
 25     private static $changed = false;
 26 
 27     /**
 28      * data read from database.
 29      *
 30      * @var array
 31      */
 32     private static $data = [];
 33 
 34     /**
 35      * data which is modified during this request.
 36      *
 37      * @var array
 38      */
 39     private static $changedData = [];
 40 
 41     /**
 42      * data which was deleted during this request.
 43      *
 44      * @var array
 45      */
 46     private static $deletedData = [];
 47 
 48     /**
 49      * Method which saves an arbitary value associated to the given namespace and key.
 50      * If the second parameter is an associative array, all key/value pairs will be saved.
 51      *
 52      * The set-method returns TRUE when an existing value was overridden, otherwise FALSE is returned.
 53      *
 54      * @param string       $namespace The namespace e.g. an addon name
 55      * @param string|array $key       The associated key or an associative array of key/value pairs
 56      * @param mixed        $value     The value to save
 57      *
 58      * @throws InvalidArgumentException
 59      *
 60      * @return bool TRUE when an existing value was overridden, otherwise FALSE
 61      */
 62     public static function set($namespace, $key, $value = null)
 63     {
 64         self::init();
 65 
 66         if (!is_string($namespace)) {
 67             throw new InvalidArgumentException('rex_config: expecting $namespace to be a string, ' . gettype($namespace) . ' given!');
 68         }
 69 
 70         if (is_array($key)) {
 71             $existed = false;
 72             foreach ($key as $k => $v) {
 73                 $existed = self::set($namespace, $k, $v) || $existed;
 74             }
 75             return $existed;
 76         }
 77 
 78         if (!is_string($key)) {
 79             throw new InvalidArgumentException('rex_config: expecting $key to be a string, ' . gettype($key) . ' given!');
 80         }
 81 
 82         if (!isset(self::$data[$namespace])) {
 83             self::$data[$namespace] = [];
 84         }
 85 
 86         $existed = isset(self::$data[$namespace][$key]);
 87         if (!$existed || $existed && self::$data[$namespace][$key] !== $value) {
 88             // keep track of changed data
 89             self::$changedData[$namespace][$key] = $value;
 90 
 91             // since it was re-added, do not longer mark as deleted
 92             unset(self::$deletedData[$namespace][$key]);
 93 
 94             // re-set the data in the container
 95             self::$data[$namespace][$key] = $value;
 96             self::$changed = true;
 97         }
 98 
 99         return $existed;
100     }
101 
102     /**
103      * Method which returns an associated value for the given namespace and key.
104      * If $key is null, an array of all key/value pairs for the given namespace will be returned.
105      *
106      * If no value can be found for the given key/namespace combination $default is returned.
107      *
108      * @param string $namespace The namespace e.g. an addon name
109      * @param string $key       The associated key
110      * @param mixed  $default   Default return value if no associated-value can be found
111      *
112      * @throws InvalidArgumentException
113      *
114      * @return mixed the value for $key or $default if $key cannot be found in the given $namespace
115      */
116     public static function get($namespace, $key = null, $default = null)
117     {
118         self::init();
119 
120         if (!is_string($namespace)) {
121             throw new InvalidArgumentException('rex_config: expecting $namespace to be a string, ' . gettype($namespace) . ' given!');
122         }
123 
124         if ($key === null) {
125             return isset(self::$data[$namespace]) ? self::$data[$namespace] : [];
126         }
127 
128         if (!is_string($key)) {
129             throw new InvalidArgumentException('rex_config: expecting $key to be a string, ' . gettype($key) . ' given!');
130         }
131 
132         if (isset(self::$data[$namespace][$key])) {
133             return self::$data[$namespace][$key];
134         }
135         return $default;
136     }
137 
138     /**
139      * Returns if the given key is set.
140      *
141      * @param string $namespace The namespace e.g. an addon name
142      * @param string $key       The associated key
143      *
144      * @throws InvalidArgumentException
145      *
146      * @return bool TRUE if the key is set, otherwise FALSE
147      */
148     public static function has($namespace, $key = null)
149     {
150         self::init();
151 
152         if (!is_string($namespace)) {
153             throw new InvalidArgumentException('rex_config: expecting $namespace to be a string, ' . gettype($namespace) . ' given!');
154         }
155 
156         if ($key === null) {
157             return isset(self::$data[$namespace]);
158         }
159 
160         if (!is_string($key)) {
161             throw new InvalidArgumentException('rex_config: expecting $key to be a string, ' . gettype($key) . ' given!');
162         }
163 
164         return isset(self::$data[$namespace][$key]);
165     }
166 
167     /**
168      * Removes the setting associated with the given namespace and key.
169      *
170      * @param string $namespace The namespace e.g. an addon name
171      * @param string $key       The associated key
172      *
173      * @throws InvalidArgumentException
174      *
175      * @return bool TRUE if the value was found and removed, otherwise FALSE
176      */
177     public static function remove($namespace, $key)
178     {
179         self::init();
180 
181         if (!is_string($namespace)) {
182             throw new InvalidArgumentException('rex_config: expecting $namespace to be a string, ' . gettype($namespace) . ' given!');
183         }
184         if (!is_string($key)) {
185             throw new InvalidArgumentException('rex_config: expecting $key to be a string, ' . gettype($key) . ' given!');
186         }
187 
188         if (isset(self::$data[$namespace][$key])) {
189             // keep track of deleted data
190             self::$deletedData[$namespace][$key] = true;
191 
192             // since it will be deleted, do not longer mark as changed
193             unset(self::$changedData[$namespace][$key]);
194 
195             // delete the data from the container
196             unset(self::$data[$namespace][$key]);
197             if (empty(self::$data[$namespace])) {
198                 unset(self::$data[$namespace]);
199             }
200             self::$changed = true;
201             return true;
202         }
203         return false;
204     }
205 
206     /**
207      * Removes all settings associated with the given namespace.
208      *
209      * @param string $namespace The namespace e.g. an addon name
210      *
211      * @throws InvalidArgumentException
212      *
213      * @return bool TRUE if the namespace was found and removed, otherwise FALSE
214      */
215     public static function removeNamespace($namespace)
216     {
217         self::init();
218 
219         if (!is_string($namespace)) {
220             throw new InvalidArgumentException('rex_config: expecting $namespace to be a string, ' . gettype($namespace) . ' given!');
221         }
222 
223         if (isset(self::$data[$namespace])) {
224             foreach (self::$data[$namespace] as $key => $value) {
225                 self::remove($namespace, $key);
226             }
227 
228             unset(self::$data[$namespace]);
229             self::$changed = true;
230             return true;
231         }
232         return false;
233     }
234 
235     /**
236      * Refreshes rex_config by reloading config from db.
237      */
238     public static function refresh()
239     {
240         if (!self::$initialized) {
241             self::init();
242 
243             return;
244         }
245 
246         self::loadFromDb();
247 
248         self::generateCache();
249 
250         self::$changed = false;
251         self::$changedData = [];
252         self::$deletedData = [];
253     }
254 
255     /**
256      * initilizes the rex_config class.
257      */
258     protected static function init()
259     {
260         if (self::$initialized) {
261             return;
262         }
263 
264         define('REX_CONFIG_FILE_CACHE', rex_path::coreCache('config.cache'));
265 
266         // take care, so we are able to write a cache file on shutdown
267         // (check here, since exceptions in shutdown functions are not visible to the user)
268         $dir = dirname(REX_CONFIG_FILE_CACHE);
269         rex_dir::create($dir);
270         if (!is_writable($dir)) {
271             throw new rex_exception('rex-config: cache dir "' . dirname(REX_CONFIG_FILE_CACHE) . '" is not writable!');
272         }
273 
274         // save cache on shutdown
275         register_shutdown_function([__CLASS__, 'save']);
276 
277         self::load();
278         self::$initialized = true;
279     }
280 
281     /**
282      * load the config-data.
283      */
284     protected static function load()
285     {
286         // check if we can load the config from the filesystem
287         if (!self::loadFromFile()) {
288             // if not possible, fallback to load config from the db
289             self::loadFromDb();
290             // afterwards persist loaded data into file-cache
291             self::generateCache();
292         }
293     }
294 
295     /**
296      * load the config-data from a file-cache.
297      *
298      * @return bool Returns TRUE, if the data was successfully loaded from the file-cache, otherwise FALSE
299      */
300     private static function loadFromFile()
301     {
302         // delete cache-file, will be regenerated on next request
303         if (file_exists(REX_CONFIG_FILE_CACHE)) {
304             self::$data = rex_file::getCache(REX_CONFIG_FILE_CACHE);
305             return true;
306         }
307         return false;
308     }
309 
310     /**
311      * load the config-data from database.
312      */
313     private static function loadFromDb()
314     {
315         $sql = rex_sql::factory();
316         $sql->setQuery('SELECT * FROM ' . rex::getTablePrefix() . 'config');
317 
318         self::$data = [];
319         foreach ($sql as $cfg) {
320             self::$data[$cfg->getValue('namespace')][$cfg->getValue('key')] = json_decode($cfg->getValue('value'), true);
321         }
322     }
323 
324     /**
325      * save config to file-cache.
326      */
327     private static function generateCache()
328     {
329         if (rex_file::putCache(REX_CONFIG_FILE_CACHE, self::$data) <= 0) {
330             throw new rex_exception('rex-config: unable to write cache file ' . REX_CONFIG_FILE_CACHE);
331         }
332     }
333 
334     /**
335      * persists the config-data and truncates the file-cache.
336      */
337     public static function save()
338     {
339         // save cache only if changes happened
340         if (!self::$changed) {
341             return;
342         }
343 
344         // after all no data needs to be deleted or update, so skip save
345         if (empty(self::$deletedData) && empty(self::$changedData)) {
346             return;
347         }
348 
349         // delete cache-file; will be regenerated on next request
350         rex_file::delete(REX_CONFIG_FILE_CACHE);
351 
352         // save all data to the db
353         self::saveToDb();
354         self::$changed = false;
355         self::$changedData = [];
356         self::$deletedData = [];
357     }
358 
359     /**
360      * save the config-data into the db.
361      */
362     private static function saveToDb()
363     {
364         $sql = rex_sql::factory();
365         // $sql->setDebug();
366 
367         // remove all deleted data
368         foreach (self::$deletedData as $namespace => $nsData) {
369             foreach ($nsData as $key => $value) {
370                 $sql->setTable(rex::getTablePrefix() . 'config');
371                 $sql->setWhere([
372                     'namespace' => $namespace,
373                     'key' => $key,
374                 ]);
375                 $sql->delete();
376             }
377         }
378 
379         // update all changed data
380         foreach (self::$changedData as $namespace => $nsData) {
381             foreach ($nsData as $key => $value) {
382                 $sql->setTable(rex::getTablePrefix() . 'config');
383                 $sql->setValue('namespace', $namespace);
384                 $sql->setValue('key', $key);
385                 $sql->setValue('value', json_encode($value));
386                 $sql->replace();
387             }
388         }
389     }
