 92     /**
 93      * select style
 94      * Es ist moeglich sowohl eine Styleklasse als auch einen Style zu uebergeben.
 95      *
 96      * Aufrufbeispiel:
 97      * $sel_media->setStyle('class="inp100"');
 98      * und/oder
 99      * $sel_media->setStyle("width:150px;");
100      */
101     public function setStyle($style)
102     {
103         if (strpos($style, 'class=') !== false) {
104             if (preg_match('/class=["\']?([^"\']*)["\']?/i', $style, $matches)) {
105                 $this->setAttribute('class', $matches[1]);
106             }
107         } else {
108             $this->setAttribute('style', $style);
109         }
110     }

148     /**
149      * Fügt ein Array von Optionen hinzu, dass eine mehrdimensionale Struktur hat.
150      *
151      * Dim   Wert
152      * 0.    Name
153      * 1.    Value
154      * 2.    Id
155      * 3.    parent_id
156      * 4.    Selected
157      * 5.    Attributes
158      */
159     public function addOptions($options, $useOnlyValues = false)
160     {
161         if (is_array($options) && count($options) > 0) {
162             // Hier vorher auf is_array abfragen, da bei Strings auch die Syntax mit [] funktioniert
163             // $ab = "hallo"; $ab[2] -> "l"
164             $grouped = isset($options[0]) && is_array($options[0]) && isset($options[0][2]) && isset($options[0][3]);
165             foreach ($options as $key => $option) {
166                 $option = (array) $option;
167                 $attributes = [];
168                 if (isset($option[5]) && is_array($option[5])) {
169                     $attributes = $option[5];
170                 }
171                 if ($grouped) {
172                     $this->addOption($option[0], $option[1], $option[2], $option[3], $attributes);
173                     if (isset($option[4]) && $option[4]) {
174                         $this->setSelected($option[1]);
175                     }
176                 } else {
177                     if ($useOnlyValues) {
178                         $this->addOption($option[0], $option[0]);
179                     } else {
180                         if (!isset($option[1])) {
181                             $option[1] = $key;
182                         }
183 
184                         $this->addOption($option[0], $option[1]);
185                     }
186                 }
187             }
188         }
189     }

195     public function addArrayOptions(array $options, $use_keys = true)
196     {

229     public function get()
230     {
231         $attr = '';
232         foreach ($this->attributes as $name => $value) {
233             $attr .= ' ' . $name . '="' . $value . '"';
234         }
235 
236         $ausgabe = "\n";
237         $ausgabe .= '<select' . $attr . '>' . "\n";
238 
239         foreach ($this->options as $optgroup => $options) {
240             $this->currentOptgroup = $optgroup;
241             if ($optgroupLabel = isset($this->optgroups[$optgroup]) ? $this->optgroups[$optgroup] : null) {
242                 $ausgabe .= '  <optgroup label="' . rex_escape($optgroupLabel) . '">' . "\n";
243             }
244             if (is_array($options)) {
245                 $ausgabe .= $this->outGroup(0);
246             }
247             if ($optgroupLabel) {
248                 $ausgabe .= '  </optgroup>' . "\n";
249             }
250         }
251 
252         $ausgabe .= '</select>' . "\n";
253         return $ausgabe;
254     }

261     protected function outGroup($parent_id, $level = 0)
262     {
263         if ($level > 100) {
264             // nur mal so zu sicherheit .. man weiss nie ;)
265             throw new rex_exception('rex_select->outGroup overflow');
266         }
267 
268         $ausgabe = '';
269         $group = $this->getGroup($parent_id);
270         if (!is_array($group)) {
271             return '';
272         }
273         foreach ($group as $option) {
274             $name = $option[0];
275             $value = $option[1];
276             $id = $option[2];
277             $attributes = [];
278             if (isset($option[3]) && is_array($option[3])) {
279                 $attributes = $option[3];
280             }
281             $ausgabe .= $this->outOption($name, $value, $level, $attributes);
282 
283             $subgroup = $this->getGroup($id, true);
284             if ($subgroup !== false) {
285                 $ausgabe .= $this->outGroup($id, $level + 1);
286             }
287         }
288         return $ausgabe;
289     }
290 
291     protected function outOption($name, $value, $level = 0, array $attributes = [])
292     {
293         $name = rex_escape($name);
294         // for BC reasons, we always expect value to be a string.
295         // this also makes sure that the strict in_array() check below works.
296         $value = (string) rex_escape($value);
297 
298         $bsps = '';
299         if ($level > 0) {
300             $bsps = str_repeat('&nbsp;&nbsp;&nbsp;', $level);
301         }
302 
303         if ($this->option_selected !== null && in_array($value, $this->option_selected, true)) {
304             $attributes['selected'] = 'selected';
305         }
306 
307         $attr = '';
308         foreach ($attributes as $n => $v) {
309             $attr .= ' ' . $n . '="' . $v . '"';
310         }
311 
312         return '        <option value="' . $value . '"' . $attr . '>' . $bsps . $name . '</option>' . "\n";
313     }
314 
315     protected function getGroup($parent_id, $ignore_main_group = false)
316     {
317         if ($ignore_main_group && $parent_id == 0) {
318             return false;
319         }
320 
321         if (isset($this->options[$this->currentOptgroup][$parent_id])) {
322             return $this->options[$this->currentOptgroup][$parent_id];
323         }
324 
325         return false;
326     }
