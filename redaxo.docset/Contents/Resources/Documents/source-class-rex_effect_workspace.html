 53     public function execute()
 54     {
 55         $this->media->asImage();
 56 
 57         $gdimage = $this->media->getImage();
 58         $w = $this->media->getWidth();
 59         $h = $this->media->getHeight();
 60 
 61         $this->params['width'] = (int) $this->params['width'];
 62         if ($this->params['width'] <= 0) {
 63             $this->params['width'] = $w;
 64         }
 65 
 66         $this->params['height'] = (int) $this->params['height'];
 67         if ($this->params['height'] <= 0) {
 68             $this->params['height'] = $h;
 69         }
 70 
 71         $this->params['bg_r'] = (int) $this->params['bg_r'];
 72         if (!isset($this->params['bg_r']) || $this->params['bg_r'] > 255 || $this->params['bg_r'] < 0) {
 73             $this->params['bg_r'] = 255;
 74         }
 75 
 76         $this->params['bg_g'] = (int) $this->params['bg_g'];
 77         if (!isset($this->params['bg_g']) || $this->params['bg_g'] > 255 || $this->params['bg_g'] < 0) {
 78             $this->params['bg_g'] = 255;
 79         }
 80 
 81         $this->params['bg_b'] = (int) $this->params['bg_b'];
 82         if (!isset($this->params['bg_b']) || $this->params['bg_b'] > 255 || $this->params['bg_b'] < 0) {
 83             $this->params['bg_b'] = 255;
 84         }
 85 
 86         $trans = false;
 87         if ($this->params['set_transparent'] != 'colored') {
 88             if ($this->media->getFormat() != 'gif' && $this->media->getFormat() != 'png' && $this->media->getFormat() != 'webp') {
 89                 $this->media->setFormat('png');
 90             }
 91             $trans = true;
 92         }
 93 
 94         $workspace = imagecreatetruecolor($this->params['width'], $this->params['height']);
 95         if ($trans) {
 96             imagealphablending($workspace, false);
 97             $transparent = imagecolorallocatealpha($workspace, 0, 0, 0, 127);
 98             imagefill($workspace, 0, 0, $transparent);
 99             imagesavealpha($workspace, true);
100             imagealphablending($workspace, true);
101         } else {
102             imagefill($workspace, 0, 0, imagecolorallocate($workspace, $this->params['bg_r'], $this->params['bg_g'], $this->params['bg_b']));
103         }
104 
105         $src_w = $w;
106         $src_h = $h;
107         $dst_x = 0;
108         $dst_y = 0;
109         $src_x = 0;
110         $src_y = 0;
111 
112         switch ($this->params['vpos']) {
113             case 'top':
114                 break;
115             case 'bottom':
116                 $dst_y = (int) $this->params['height'] - $h;
117                 break;
118             case 'middle':
119             default: // center
120                 $dst_y = (int) ($this->params['height'] / 2) - ($h / 2);
121                 break;
122         }
123 
124         switch ($this->params['hpos']) {
125             case 'left':
126                 break;
127             case 'right':
128                 $dst_x = (int) $this->params['width'] - $w;
129                 break;
130             case 'center':
131             default: // center
132                 $dst_x = (int) ($this->params['width'] / 2) - ($w / 2);
133                 break;
134         }
135 
136         imagecopy($workspace, $gdimage, $dst_x, $dst_y, $src_x, $src_y, $src_w, $src_h);
137         $this->media->setImage($workspace);
138         $this->media->refreshImageDimensions();
139 
140         // Transparenz erhalten
141         /*
142         $this->keepTransparent($des);
143         imagecopyresampled($des, $gdimage, 0, 0, 0, 0, $this->params['width'], $this->params['height'], $w, $h);
144 
145         $gdimage = $des;
146         $this->image->refreshDimensions();
147         */
148     }
