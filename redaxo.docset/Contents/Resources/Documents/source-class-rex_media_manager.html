 24     /**
 25      * Creates a rex_managed_media object for the given file and mediatype.
 26      * This object might be used to determine the dimension of a image or similar.
 27      *
 28      * @param string $type Media type
 29      * @param string $file Media file
 30      *
 31      * @return self
 32      */
 33     public static function create($type, $file)
 34     {
 35         $mediaPath = rex_path::media($file);
 36         $cachePath = rex_path::addonCache('media_manager');
 37 
 38         $media = new rex_managed_media($mediaPath);
 39         $manager = new self($media);
 40         $manager->setCachePath($cachePath);
 41         $manager->applyEffects($type);
 42 
 43         if ($manager->isCached()) {
 44             $media->setSourcePath($manager->getCacheFilename());
 45 
 46             $cache = $manager->getHeaderCache();
 47 
 48             $media->setFormat($cache['format']);
 49 
 50             foreach ($cache['headers'] as $key => $value) {
 51                 $media->setHeader($key, $value);
 52             }
 53         } else {
 54             $media->save($manager->getCacheFilename(), $manager->getHeaderCacheFilename());
 55         }
 56 
 57         $media->refreshImageDimensions();
 58 
 59         return $manager;
 60     }

 70     protected function applyEffects($type)
 71     {
 72         $this->type = $type;
 73 
 74         if (!$this->isCached()) {
 75             $set = $this->effectsFromType($type);
 76             $set = rex_extension::registerPoint(new rex_extension_point('MEDIA_MANAGER_FILTERSET', $set, ['rex_media_type' => $type]));
 77 
 78             if (count($set) == 0) {
 79                 return $this->media;
 80             }
 81 
 82             // execute effects on image
 83             foreach ($set as $effect_params) {
 84                 $effect_class = 'rex_effect_' . $effect_params['effect'];
 85                 /** @var rex_effect_abstract $effect */
 86                 $effect = new $effect_class();
 87                 $effect->setMedia($this->media);
 88                 $effect->setParams($effect_params['params']);
 89                 $effect->execute();
 90             }
 91         }
 92     }

 94     public function effectsFromType($type)
 95     {
 96         $qry = '
 97             SELECT e.*
 98             FROM ' . rex::getTablePrefix() . 'media_manager_type t, ' . rex::getTablePrefix() . 'media_manager_type_effect e
 99             WHERE e.type_id = t.id AND t.name=? order by e.priority';
100 
101         $sql = rex_sql::factory();
102         // $sql->setDebug();
103         $sql->setQuery($qry, [$type]);
104 
105         $effects = [];
106         foreach ($sql as $row) {
107             $effname = $row->getValue('effect');
108             $params = json_decode($row->getValue('parameters'), true);
109             $effparams = [];
110 
111             // extract parameter out of array
112             if (isset($params['rex_effect_' . $effname])) {
113                 foreach ($params['rex_effect_' . $effname] as $name => $value) {
114                     $effparams[str_replace('rex_effect_' . $effname . '_', '', $name)] = $value;
115                     unset($effparams[$name]);
116                 }
117             }
118 
119             $effect = [
120                 'effect' => $effname,
121                 'params' => $effparams,
122             ];
123 
124             $effects[] = $effect;
125         }
126 
127         return $effects;
128     }

145     public function isCached()
146     {
147         $cache_file = $this->getCacheFilename();
148 
149         if (!file_exists($cache_file)) {
150             return false;
151         }
152 
153         $cache = $this->getHeaderCache();
154 
155         if (!$cache) {
156             return false;
157         }
158 
159         $mediapath = $cache['media_path'];
160 
161         if (null === $mediapath) {
162             return true;
163         }
164 
165         if (!file_exists($mediapath)) {
166             return false;
167         }
168 
169         $cachetime = filemtime($cache_file);
170         $filetime = filemtime($mediapath);
171 
172         // cache is newer?
173         return $cachetime > $filetime;
174     }

195     public static function deleteCacheByType($type_id)
196     {
197         $qry = 'SELECT * FROM ' . rex::getTablePrefix() . 'media_manager_type' . ' WHERE id=?';
198         $sql = rex_sql::factory();
199         //  $sql->setDebug();
200         $sql->setQuery($qry, [$type_id]);
201         $counter = 0;
202         foreach ($sql as $row) {
203             $counter += self::deleteCache(null, $row->getValue('name'));
204         }
205         return $counter;
206     }

208     public static function deleteCache($filename = null, $type = null)
209     {
210         $filename = ($filename ?: '').'*';
211 
212         if (!$type) {
213             $type = '*';
214         }
215 
216         $counter = 0;
217         $folder = rex_path::addonCache('media_manager');
218 
219         $glob = glob($folder.$type.'/'.$filename, GLOB_NOSORT);
220         if ($glob) {
221             foreach ($glob as $file) {
222                 if (rex_file::delete($file)) {
223                     ++$counter;
224                 }
225             }
226         }
227 
228         return $counter;
229     }

231     public function sendMedia()
232     {
233         $headerCacheFilename = $this->getHeaderCacheFilename();
234         $CacheFilename = $this->getCacheFilename();
235 
236         rex_response::cleanOutputBuffers();
237 
238         // prevent session locking trough other addons
239         if (function_exists('session_abort')) {
240             session_abort();
241         } else {
242             session_write_close();
243         }
244 
245         if ($this->isCached()) {
246             $header = $this->getHeaderCache()['headers'];
247             if (isset($header['Last-Modified'])) {
248                 rex_response::sendLastModified(strtotime($header['Last-Modified']));
249                 unset($header['Last-Modified']);
250             }
251             foreach ($header as $t => $c) {
252                 header($t . ': ' . $c);
253             }
254             readfile($CacheFilename);
255         } else {
256             $this->media->sendMedia($CacheFilename, $headerCacheFilename, $this->use_cache);
257         }
258         exit;
259     }

261     public static function getSupportedEffects()
262     {
263         $dirs = [
264             __DIR__ . '/effects/',
265         ];
266 
267         $effects = [];
268         foreach ($dirs as $dir) {
269             $files = glob($dir . 'effect_*.php');
270             if ($files) {
271                 foreach ($files as $file) {
272                     $effects[self::getEffectClass($file)] = self::getEffectName($file);
273                 }
274             }
275         }
276 
277         foreach (self::$effects as $class) {
278             $effects[$class] = str_replace(['rex_', 'effect_'], '', $class);
279         }
280 
281         return $effects;
282     }

315     public static function init()
316     {
317         //--- handle image request
318         $rex_media_manager_file = self::getMediaFile();
319         $rex_media_manager_type = self::getMediaType();
320 
321         if ($rex_media_manager_file != '' && $rex_media_manager_type != '') {
322             $media_path = rex_path::media($rex_media_manager_file);
323             $cache_path = rex_path::addonCache('media_manager');
324 
325             $media = new rex_managed_media($media_path);
326             $media_manager = new self($media);
327             $media_manager->setCachePath($cache_path);
328             $media_manager->applyEffects($rex_media_manager_type);
329             $media_manager->sendMedia();
330 
331             exit();
332         }
333     }
