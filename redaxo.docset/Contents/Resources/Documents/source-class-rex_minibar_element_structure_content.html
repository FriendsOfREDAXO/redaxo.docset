1 <?php
  2 
  3 /**
  4  * @package redaxo\core
  5  */
  6 class rex_minibar_element_structure_content extends rex_minibar_element
  7 {
  8     /**
  9      * {@inheritdoc}
 10      */
 11     public function render()
 12     {
 13         $clangId = rex_request('clang', 'int');
 14         $clangId = rex_clang::exists($clangId) ? $clangId : rex_clang::getStartId();
 15 
 16         $article = rex_article::getCurrent();
 17         if (rex::isBackend()) {
 18             $article = rex_article::get(rex_request('article_id', 'int'), $clangId);
 19 
 20             if (!$article) {
 21                 $article = rex_article::get(rex_request('category_id', 'int'), $clangId);
 22             }
 23         }
 24 
 25         if (!$article) {
 26             $article = rex_article::getSiteStartArticle();
 27         }
 28 
 29         if (!$article instanceof rex_article) {
 30             return '';
 31         }
 32 
 33         // Return if user have no rights to the site start article
 34         if (rex::isBackend() && rex::getUser() && !rex::getUser()->getComplexPerm('structure')->hasCategoryPerm($article->getCategoryId())) {
 35             return
 36                 '<div class="rex-minibar-item">
 37                     <span class="rex-minibar-icon">
 38                         <i class="rex-icon rex-icon-article"></i>
 39                     </span>
 40                     <span class="rex-minibar-value">
 41                         0
 42                     </span>
 43                 </div>';
 44         }
 45 
 46         $articleLink = '<a href="'.rex_url::backendPage('content/edit', ['article_id' => $article->getId(), 'clang' => $article->getClangId(), 'mode' => 'edit']).'">'.rex_i18n::msg('structure_content_minibar_article_edit').' </a>';
 47         if (rex::getUser() && !rex::getUser()->getComplexPerm('structure')->hasCategoryPerm($article->getCategoryId())) {
 48             $articleLink = rex_i18n::msg('no_rights_to_edit');
 49         } elseif (rex::isBackend()) {
 50             $articleLink = '<a href="'.rex_getUrl($article->getId(), $article->getClangId()).'">'.rex_i18n::msg('structure_content_minibar_article_show').'</a>';
 51         }
 52 
 53         $articlePath = [];
 54         $tree = $article->getParentTree();
 55         if (!$article->isStartarticle()) {
 56             $tree[] = $article;
 57         }
 58         foreach ($tree as $parent) {
 59             $id = $parent->getId();
 60             $item = rex_escape($parent->getName());
 61             if (rex::isBackend() && rex::getUser() && rex::getUser()->getComplexPerm('structure')->hasCategoryPerm($id) && $parent->isStartarticle()) {
 62                 $item = '<a href="'.rex_url::backendPage('structure', ['category_id' => $id, 'clang' => $clangId]).'">'.rex_escape($parent->getName()).'</a>';
 63             } elseif (!rex::isBackend()) {
 64                 $item = '<a href="'.$parent->getUrl().'">'.rex_escape($parent->getName()).'</a>';
 65             }
 66             $articlePath[] = $item;
 67         }
 68 
 69         return
 70         '<div class="rex-minibar-item">
 71             <span class="rex-minibar-icon">
 72                 <i class="rex-minibar-icon--fa rex-minibar-icon--fa-file-text-o"></i>
 73             </span>
 74             <span class="rex-minibar-value">
 75                 '.$article->getId().'
 76             </span>
 77         </div>
 78         <div class="rex-minibar-info">
 79             <div class="rex-minibar-info-group">
 80                 <div class="rex-minibar-info-piece">
 81                     <b>'.rex_i18n::msg('structure_article_name').'</b>
 82                     <span>'.rex_escape($article->getName()).'</span>
 83                 </div>
 84                 <div class="rex-minibar-info-piece">
 85                     <b>'.rex_i18n::msg('structure_status').'</b>
 86                     <span class="rex-minibar-status-'.($article->isOnline() ? 'green' : 'red').'">'.($article->isOnline() ? rex_i18n::msg('status_online') : rex_i18n::msg('status_offline')).'</span>                    
 87                 </div>
 88                 <div class="rex-minibar-info-piece">
 89                     <b>'.rex_i18n::msg('structure_path').'</b>
 90                     <span>'.implode(' / ', $articlePath).'</span>
 91                 </div>
 92                 <div class="rex-minibar-info-piece">
 93                     <b></b>
 94                     <span>'.$articleLink.'</span>
 95                 </div>
 96             </div>
 97         </div>
 98         ';
 99     }
100 }
