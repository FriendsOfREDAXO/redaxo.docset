 32     /**
 33      * Diese Konstruktor sollte nicht verwendet werden. Instanzen muessen ueber die facotry() Methode erstellt werden!
 34      */
 35     protected function __construct($tableName, $fieldset, $whereCondition, $method = 'post', $debug = false)
 36     {
 37         $name = md5($tableName . $whereCondition . $method);
 38 
 39         parent::__construct($fieldset, $name, $method, $debug);
 40 
 41         $this->tableName = $tableName;
 42         $this->whereCondition = $whereCondition;
 43         $this->languageSupport = [];
 44 
 45         $this->sql = rex_sql::factory();
 46         $this->sql->setDebug($this->debug);
 47         $this->sql->setQuery('SELECT * FROM ' . $tableName . ' WHERE ' . $this->whereCondition . ' LIMIT 2');
 48 
 49         $this->setFormId('rex-addon-editmode');
 50 
 51         // --------- validate where-condition and determine editMode
 52         $numRows = $this->sql->getRows();
 53         if ($numRows == 0) {
 54             // Kein Datensatz gefunden => Mode: Add
 55             $this->setEditMode(false);
 56         } elseif ($numRows == 1) {
 57             // Ein Datensatz gefunden => Mode: Edit
 58             $this->setEditMode(true);
 59         } else {
 60             throw new rex_exception('rex_form: Die gegebene Where-Bedingung f√ºhrt nicht zu einem eindeutigen Datensatz!');
 61         }
 62 
 63         // --------- Load Env
 64         if (rex::isBackend()) {
 65             $this->loadBackendConfig();
 66         }
 67     }

 69     /**
 70      * Methode zum erstellen von rex_form Instanzen.

 72      * @param string $tableName

 80     public static function factory($tableName, $fieldset, $whereCondition, $method = 'post', $debug = false)
 81     {
 82         $class = static::getFactoryClass();
 83         return new $class($tableName, $fieldset, $whereCondition, $method, $debug);
 84     }

 86     /**
 87      * Laedt die Konfiguration die noetig ist um rex_form im REDAXO Backend zu verwenden.
 88      */
 89     protected function loadBackendConfig()
 90     {
 91         parent::loadBackendConfig();
 92 
 93         $func = rex_request('func', 'string');
 94 
 95         $this->addParam('func', $func);
 96         $this->addParam('list', rex_request('list', 'string'));
 97 
 98         $controlFields = [];
 99         $controlFields['save'] = rex_i18n::msg('form_save');
100         $controlFields['apply'] = $func == 'edit' ? rex_i18n::msg('form_apply') : '';
101         $controlFields['delete'] = $func == 'edit' ? rex_i18n::msg('form_delete') : '';
102         $controlFields['reset'] = ''; //rex_i18n::msg('form_reset');
103         $controlFields['abort'] = rex_i18n::msg('form_abort');
104 
105         // ----- EXTENSION POINT
106         $controlFields = rex_extension::registerPoint(new rex_extension_point('REX_FORM_CONTROL_FIELDS', $controlFields, ['form' => $this]));
107 
108         $controlElements = [];
109         foreach ($controlFields as $name => $label) {
110             if ($label) {
111                 $attr = ['type' => 'submit', 'internal::useArraySyntax' => false, 'internal::fieldSeparateEnding' => true];
112 
113                 if ($name === 'abort' || $name === 'delete') {
114                     $attr['formnovalidate'] = 'formnovalidate';
115                 }
116                 $controlElements[$name] = $this->addField(
117                     'button',
118                     $name,
119                     $label,
120                     $attr,
121                     false
122                 );
123             } else {
124                 $controlElements[$name] = null;
125             }
126         }
127 
128         $this->addControlField(
129             $controlElements['save'],
130             $controlElements['apply'],
131             $controlElements['delete'],
132             $controlElements['reset'],
133             $controlElements['abort']
134         );
135     }

146     public function addPrioField($name, $value = null, array $attributes = [])
147     {
148         $attributes['internal::fieldClass'] = 'rex_form_prio_element';

228     /**
229      * Callbackfunktion, damit in subklassen der Value noch beeinflusst werden kann
230      * kurz vorm speichern.
231      */
232     protected function preSave($fieldsetName, $fieldName, $fieldValue, rex_sql $saveSql)
233     {
234         static $setOnce = false;
235 
236         if (!$setOnce) {
237             $fieldnames = $this->sql->getFieldnames();
238 
239             if (in_array('updateuser', $fieldnames)) {
240                 $saveSql->setValue('updateuser', rex::getUser()->getValue('login'));
241             }
242 
243             if (in_array('updatedate', $fieldnames)) {
244                 $saveSql->setDateTimeValue('updatedate', time());
245             }
246 
247             if (!$this->isEditMode()) {
248                 if (in_array('createuser', $fieldnames)) {
249                     $saveSql->setValue('createuser', rex::getUser()->getValue('login'));
250                 }
251 
252                 if (in_array('createdate', $fieldnames)) {
253                     $saveSql->setDateTimeValue('createdate', time());
254                 }
255             }
256             $setOnce = true;
257         }
258 
259         return $fieldValue;
260     }

275     /**
276      * Speichert das Formular.
277      *
278      * √úbernimmt die Werte aus den FormElementen in die Datenbank.
279      *
280      * Gibt true zur√ºck wenn alles ok war, false bei einem allgemeinen Fehler,
281      * einen String mit einer Fehlermeldung oder den von der Datenbank gelieferten ErrorCode.
282      *
283      * @return bool
284      */
285     protected function save()
286     {
287         $sql = rex_sql::factory();
288         $sql->setDebug($this->debug);
289         $sql->setTable($this->tableName);
290 
291         $values = [];
292         foreach ($this->getSaveElements() as $fieldsetName => $fieldsetElements) {
293             foreach ($fieldsetElements as $element) {
294                 // read-only-fields nicht speichern
295                 if (strpos($element->getAttribute('class'), 'form-control-static') !== false) {
296                     continue;
297                 }
298 
299                 $fieldName = $element->getFieldName();
300                 $fieldValue = $element->getSaveValue();
301 
302                 // Callback, um die Values vor dem Speichern noch beeinflussen zu k√∂nnen
303                 $fieldValue = $this->preSave($fieldsetName, $fieldName, $fieldValue, $sql);
304 
305                 if (is_string($fieldValue)) {
306                     $fieldValue = trim($fieldValue);
307                 }
308                 $values[$fieldName] = $fieldValue;
309             }
310         }
311 
312         try {
313             if ($this->isEditMode()) {
314                 $sql->setValues($values);
315                 $sql->setWhere($this->whereCondition);
316                 $sql->update();
317             } else {
318                 if (count($this->languageSupport)) {
319                     foreach (rex_clang::getAllIds() as $clang_id) {
320                         $sql->setTable($this->tableName);
321                         $sql->addGlobalCreateFields();
322                         $sql->addGlobalUpdateFields();
323                         if (!isset($id)) {
324                             $id = $sql->setNewId($this->languageSupport['id']);
325                         } else {
326                             $sql->setValue($this->languageSupport['id'], $id);
327                         }
328                         $sql->setValue($this->languageSupport['clang'], $clang_id);
329                         $sql->setValues($values);
330                         $sql->insert();
331                     }
332                 } else {
333                     $sql->setValues($values);
334                     $sql->insert();
335                 }
336             }
337             $saved = true;
338         } catch (rex_sql_exception $e) {
339             $saved = false;
340         }
341 
342         // ----- EXTENSION POINT
343         if ($saved) {
344             $saved = rex_extension::registerPoint(new rex_extension_point('REX_FORM_SAVED', $saved, ['form' => $this, 'sql' => $sql]));
345         } else {
346             $saved = $sql->getMysqlErrno();
347         }
348 
349         return $saved;
350     }

352     /**
353      * @return bool
354      */
355     protected function delete()
356     {
357         $deleteSql = rex_sql::factory();
358         $deleteSql->setDebug($this->debug);
359         $deleteSql->setTable($this->tableName);
360         $deleteSql->setWhere($this->whereCondition);
361 
362         try {
363             $deleteSql->delete();
364             $deleted = true;
365         } catch (rex_sql_exception $e) {
366             $deleted = false;
367         }
368 
369         // ----- EXTENSION POINT
370         if ($deleted) {
371             $deleted = rex_extension::registerPoint(new rex_extension_point('REX_FORM_DELETED', $deleted, ['form' => $this, 'sql' => $deleteSql]));
372         } else {
373             $deleted = $deleteSql->getMysqlErrno();
374         }
375 
376         return $deleted;
377     }
