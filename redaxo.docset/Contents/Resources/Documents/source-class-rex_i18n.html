  3 /**
  4  * Class for internationalization.
  5  *
  6  * @package redaxo\core
  7  */
  8 class rex_i18n
  9 {
 10     private static $locales = [];
 11     private static $directories = [];
 12     private static $loaded = false;
 13     private static $locale = null;
 14     private static $msg = [];
 15 
 16     /**
 17      * Switches the current locale.
 18      *
 19      * @param string $locale       The new locale
 20      * @param bool   $phpSetLocale When TRUE, php function setlocale() will be called
 21      *
 22      * @return string The last locale
 23      */
 24     public static function setLocale($locale, $phpSetLocale = true)
 25     {
 26         $saveLocale = self::$locale;
 27         self::$locale = $locale;
 28 
 29         self::loadAll();
 30 
 31         if ($phpSetLocale) {
 32             $locales = [];
 33             foreach (explode(',', trim(self::msg('setlocale'))) as $locale) {
 34                 $locales[] = $locale . '.UTF-8';
 35                 $locales[] = $locale . '.UTF8';
 36                 $locales[] = $locale . '.utf-8';
 37                 $locales[] = $locale . '.utf8';
 38                 $locales[] = $locale;
 39             }
 40 
 41             setlocale(LC_ALL, $locales);
 42         }
 43 
 44         return $saveLocale;
 45     }
 46 
 47     /**
 48      * Returns the current locale.
 49      *
 50      * @return string The current locale
 51      */
 52     public static function getLocale()
 53     {
 54         return self::$locale;
 55     }

 57     /**
 58      * Adds a directory with lang files.
 59      *
 60      * @param string $dir Path to the directory
 61      */
 62     public static function addDirectory($dir)
 63     {
 64         $dir = rtrim($dir, DIRECTORY_SEPARATOR);
 65 
 66         if (in_array($dir, self::$directories, true)) {
 67             return;
 68         }
 69 
 70         self::$directories[] = $dir;
 71 
 72         if (self::$loaded) {
 73             self::loadFile($dir . DIRECTORY_SEPARATOR . self::$locale . '.lang');
 74         }
 75     }
 76 
 77     /**
 78      * Returns the translation htmlspecialchared for the given key.
 79      *
 80      * @param string $key Key
 81      *
 82      * @return string Translation for the key
 83      */
 84     public static function msg($key)
 85     {
 86         return self::getMsg($key, true, func_get_args());
 87     }
 88 
 89     /**
 90      * Returns the translation for the given key.
 91      *
 92      * @param string $key Key
 93      *
 94      * @return string Translation for the key
 95      */
 96     public static function rawMsg($key)
 97     {
 98         return self::getMsg($key, false, func_get_args());
 99     }
100 
101     /**
102      * Returns the translation for the given key.
103      *
104      * @param string $key
105      * @param bool   $htmlspecialchars
106      * @param array  $args
107      *
108      * @return mixed
109      */
110     private static function getMsg($key, $htmlspecialchars, array $args)
111     {
112         if (!self::$loaded) {
113             self::loadAll();
114         }
115 
116         if (self::hasMsg($key)) {
117             $msg = self::$msg[$key];
118         } else {
119             $msg = "[translate:$key]";
120             $msg = rex_extension::registerPoint(new rex_extension_point(
121                 'I18N_MISSING_TRANSLATION',
122                 $msg,
123                 [
124                     'key' => $key,
125                     'args' => $args,
126                 ]
127             ));
128         }
129 
130         $patterns = [];
131         $replacements = [];
132         $argNum = count($args);
133         if ($argNum > 1) {
134             for ($i = 1; $i < $argNum; ++$i) {
135                 // zero indexed
136                 $patterns[] = '/\{' . ($i - 1) . '\}/';
137                 $replacements[] = $args[$i];
138             }
139         }
140 
141         $msg = preg_replace($patterns, $replacements, $msg);
142 
143         if ($htmlspecialchars) {
144             $msg = rex_escape($msg);
145             $msg = preg_replace('@&lt;(/?(?:b|i|code|kbd|var)|br ?/?)&gt;@i', '<$1>', $msg);
146         }
147 
148         return $msg;
149     }

152      * Checks if there is a translation for the given key.
153      *
154      * @param string $key Key
155      *
156      * @return bool TRUE on success, else FALSE

158     public static function hasMsg($key)

166      * @param string $key Key

169     public static function addMsg($key, $msg)

174     /**
175      * Returns the locales.
176      *
177      * @return array Array of Locales
178      */
179     public static function getLocales()
180     {
181         if (empty(self::$locales) && isset(self::$directories[0]) && is_readable(self::$directories[0])) {
182             self::$locales = [];
183 
184             foreach (rex_finder::factory(self::$directories[0])->filesOnly() as $file) {
185                 if (preg_match("/^(\w+)\.lang$/", $file->getFilename(), $matches)) {
186                     self::$locales[] = $matches[1];
187                 }
188             }
189         }
190 
191         return self::$locales;
192     }

194     /**
195      * Translates the $text, if it begins with 'translate:', else it returns $text.
196      *
197      * @param string   $text                 The text for translation
198      * @param bool     $use_htmlspecialchars Flag whether the translated text should be passed to htmlspecialchars()
199      * @param callable $i18nFunction         Function that returns the translation for the i18n key
200      *
201      * @throws InvalidArgumentException
202      *
203      * @return string Translated text
204      */
205     public static function translate($text, $use_htmlspecialchars = true, callable $i18nFunction = null)
206     {
207         if (!is_string($text)) {
208             throw new InvalidArgumentException('Expecting $text to be a String, "' . gettype($text) . '" given!');
209         }
210 
211         $tranKey = 'translate:';
212         $transKeyLen = strlen($tranKey);
213         if (substr($text, 0, $transKeyLen) == $tranKey) {
214             if (!$i18nFunction) {
215                 $i18nFunction = $use_htmlspecialchars ? 'self::msg' : 'self::rawMsg';
216             }
217             return call_user_func($i18nFunction, substr($text, $transKeyLen));
218         }
219         if ($use_htmlspecialchars) {
220             return htmlspecialchars($text);
221         }
222         return $text;
223     }
224 
225     /**
226      * Translates all array elements.
227      *
228      * @param mixed    $array                The Array of Strings for translation
229      * @param bool     $use_htmlspecialchars Flag whether the translated text should be passed to htmlspecialchars()
230      * @param callable $i18nFunction         Function that returns the translation for the i18n key
231      *
232      * @throws InvalidArgumentException
233      *
234      * @return mixed
235      */
236     public static function translateArray($array, $use_htmlspecialchars = true, callable $i18nFunction = null)
237     {
238         if (is_array($array)) {
239             foreach ($array as $key => $value) {
240                 $array[$key] = self::translateArray($value, $use_htmlspecialchars, $i18nFunction);
241             }
242             return $array;
243         }
244         if (is_string($array)) {
245             return self::translate($array, $use_htmlspecialchars, $i18nFunction);
246         }
247         if (null === $array || is_scalar($array)) {
248             return $array;
249         }
250         throw new InvalidArgumentException('Expecting $text to be a String or Array of Scalar, "' . gettype($array) . '" given!');
251     }

253     /**
254      * Loads the translation definitions of the given file.
255      *
256      * @param string $file Path to the file
257      *
258      * @return bool TRUE on success, FALSE on failure
259      */
260     private static function loadFile($file)
261     {
262         if (
263             ($content = rex_file::get($file)) &&
264             preg_match_all('/^([^=\s]+)\h*=\h*(.*)(?<=\S)/m', $content, $matches, PREG_SET_ORDER)
265         ) {
266             foreach ($matches as $match) {
267                 self::addMsg($match[1], $match[2]);
268             }
269         }
270     }
271 
272     /**
273      * Loads all translation defintions.
274      */
275     private static function loadAll()
276     {
277         self::$msg = [];
278         if (!self::$locale) {
279             self::$locale = rex::getProperty('lang');
280         }
281         foreach (self::$directories as $dir) {
282             self::loadFile($dir . DIRECTORY_SEPARATOR . self::$locale . '.lang');
283         }
284         self::$loaded = true;
285     }
