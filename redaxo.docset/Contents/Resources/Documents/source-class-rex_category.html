1 <?php
  2 
  3 /**
  4  * Object Oriented Framework: Bildet eine Kategorie der Struktur ab.
  5  *
  6  * @package redaxo\structure
  7  */
  8 class rex_category extends rex_structure_element
  9 {
 10     /**
 11      * Return the current category.
 12      *
 13      * @param int $clang
 14      *
 15      * @return self
 16      */
 17     public static function getCurrent($clang = null)
 18     {
 19         return rex_article::getCurrent($clang)->getCategory();
 20     }
 21 
 22     /**
 23      * Return a list of top level categories, ie.
 24      * categories that have no parent.
 25      * Returns an array of rex_category objects sorted by $priority.
 26      *
 27      * If $ignore_offlines is set to TRUE,
 28      * all categories with status 0 will be
 29      * excempt from this list!
 30      *
 31      * @param bool $ignoreOfflines
 32      * @param int  $clang
 33      *
 34      * @return self[]
 35      */
 36     public static function getRootCategories($ignoreOfflines = false, $clang = null)
 37     {
 38         return self::getChildElements(0, 'clist', $ignoreOfflines, $clang);
 39     }
 40 
 41     /**
 42      * {@inheritdoc}
 43      */
 44     public function getPriority()
 45     {
 46         return $this->catpriority;
 47     }
 48 
 49     /**
 50      * Return a list of all subcategories.
 51      * Returns an array of rex_category objects sorted by $priority.
 52      *
 53      * If $ignore_offlines is set to TRUE,
 54      * all categories with status 0 will be
 55      * excempt from this list!
 56      *
 57      * @param bool $ignoreOfflines
 58      *
 59      * @return self[]
 60      */
 61     public function getChildren($ignoreOfflines = false)
 62     {
 63         return self::getChildElements($this->id, 'clist', $ignoreOfflines, $this->clang_id);
 64     }
 65 
 66     /**
 67      * Returns the parent category.
 68      *
 69      * @return self
 70      */
 71     public function getParent()
 72     {
 73         return self::get($this->parent_id, $this->clang_id);
 74     }
 75 
 76     /**
 77      * Returns TRUE if this category is the direct
 78      * parent of the other category.
 79      *
 80      * @param self $otherCat
 81      *
 82      * @return bool
 83      */
 84     public function isParent(self $otherCat)
 85     {
 86         return $this->getId() == $otherCat->getParentId() &&
 87              $this->getClang() == $otherCat->getClang();
 88     }
 89 
 90     /**
 91      * Return a list of articles in this category
 92      * Returns an array of rex_article objects sorted by $priority.
 93      *
 94      * If $ignore_offlines is set to TRUE,
 95      * all articles with status 0 will be
 96      * excempt from this list!
 97      *
 98      * @param bool $ignoreOfflines
 99      *
100      * @return rex_article[]
101      */
102     public function getArticles($ignoreOfflines = false)
103     {
104         return rex_article::getChildElements($this->id, 'alist', $ignoreOfflines, $this->clang_id);
105     }
106 
107     /**
108      * Return the start article for this category.
109      *
110      * @return rex_article
111      */
112     public function getStartArticle()
113     {
114         return rex_article::get($this->id, $this->clang_id);
115     }
116 
117     /**
118      * Returns the name of the category.
119      *
120      * @return string
121      */
122     public function getName()
123     {
124         return $this->catname;
125     }
126 
127     /**
128      * Returns the path of the category.
129      *
130      * @return string
131      */
132     public function getPath()
133     {
134         return $this->path;
135     }
136 
137     /**
138      * @param string $value
139      *
140      * @return bool
141      */
142     public static function hasValue($value)
143     {
144         return parent::_hasValue($value, ['cat_']);
145     }
146 
147     /**
148      * {@inheritdoc}
149      */
150     public function isPermitted()
151     {
152         return (bool) rex_extension::registerPoint(new rex_extension_point('CAT_IS_PERMITTED', true, ['element' => $this]));
153     }
154 }
