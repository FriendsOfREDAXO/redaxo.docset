 73      * Set a file to be preload via http link header.
 74      *
 75      * @param string $file
 76      * @param string $type
 77      * @param string $mimeType
 78      */
 79     public static function preload($file, $type, $mimeType)
 80     {
 81         self::$preloadFiles[] = [
 82             'file' => $file,
 83             'type' => $type,
 84             'mimeType' => $mimeType,
 85         ];
 86     }

 88     private static function sendPreloadHeaders()
 89     {
 90         foreach (self::$preloadFiles as $preloadFile) {
 91             header('Link: <' . $preloadFile['file'] . '>; rel=preload; as=' . $preloadFile['type'] . '; type="' . $preloadFile['mimeType'].'"; crossorigin; nopush', false);
 92         }
 93     }

 95     /**
 96      * Redirects to a URL.
 97      *
 98      * NOTE: Execution will stop within this method!
 99      *
100      * @param string $url URL
101      *
102      * @throws InvalidArgumentException
103      */
104     public static function sendRedirect($url)
105     {
106         if (strpos($url, "\n") !== false) {
107             throw new InvalidArgumentException('Illegal redirect url "' . $url . '", contains newlines');
108         }
109 
110         self::cleanOutputBuffers();
111         self::sendAdditionalHeaders();
112         self::sendPreloadHeaders();
113 
114         header('HTTP/1.1 ' . self::$httpStatus);
115         header('Location: ' . $url);
116         exit;
117     }

119     /**
120      * Sends a file to client.
121      *
122      * @param string      $file               File path
123      * @param string      $contentType        Content type
124      * @param string      $contentDisposition Content disposition ("inline" or "attachment")
125      * @param null|string $filename           Custom Filename
126      */
127     public static function sendFile($file, $contentType, $contentDisposition = 'inline', $filename = null)
128     {
129         self::cleanOutputBuffers();
130 
131         if (!file_exists($file)) {
132             header('HTTP/1.1 ' . self::HTTP_NOT_FOUND);
133             exit;
134         }
135 
136         // prevent session locking while sending huge files
137         session_write_close();
138 
139         if (!$filename) {
140             $filename = basename($file);
141         }
142 
143         self::sendContentType($contentType);
144         header('Content-Disposition: ' . $contentDisposition . '; filename="' . $filename . '"');
145 
146         self::sendLastModified(filemtime($file));
147 
148         header('HTTP/1.1 ' . self::$httpStatus);
149         if (!self::$sentCacheControl) {
150             self::sendCacheControl('max-age=3600, must-revalidate, proxy-revalidate, private');
151         }
152 
153         // content length schicken, damit der browser einen ladebalken anzeigen kann
154         if (!ini_get('zlib.output_compression')) {
155             header('Content-Length: ' . filesize($file));
156         }
157 
158         self::sendAdditionalHeaders();
159         self::sendPreloadHeaders();
160 
161         readfile($file);
162     }

164     /**
165      * Sends a resource to the client.
166      *
167      * @param string      $content            Content
168      * @param null|string $contentType        Content type
169      * @param null|int    $lastModified       HTTP Last-Modified Timestamp
170      * @param null|string $etag               HTTP Cachekey to identify the cache
171      * @param null|string $contentDisposition Content disposition ("inline" or "attachment")
172      * @param null|string $filename           Filename
173      */
174     public static function sendResource($content, $contentType = null, $lastModified = null, $etag = null, $contentDisposition = null, $filename = null)
175     {
176         if ($contentDisposition) {
177             header('Content-Disposition: ' . $contentDisposition . '; filename="' . $filename . '"');
178         }
179 
180         self::sendCacheControl('max-age=3600, must-revalidate, proxy-revalidate, private');
181         self::sendContent($content, $contentType, $lastModified, $etag);
182     }

184     /**
185      * Sends a page to client.
186      *
187      * The page content can be modified by the Extension Point OUTPUT_FILTER
188      *
189      * @param string $content      Content of page
190      * @param int    $lastModified HTTP Last-Modified Timestamp
191      */
192     public static function sendPage($content, $lastModified = null)
193     {
194         // ----- EXTENSION POINT
195         $content = rex_extension::registerPoint(new rex_extension_point('OUTPUT_FILTER', $content));
196 
197         $hasShutdownExtension = rex_extension::isRegistered('RESPONSE_SHUTDOWN');
198         if ($hasShutdownExtension) {
199             header('Connection: close');
200         }
201 
202         self::sendContent($content, null, $lastModified);
203 
204         // ----- EXTENSION POINT - (read only)
205         if ($hasShutdownExtension) {
206             // unlock session
207             session_write_close();
208 
209             rex_extension::registerPoint(new rex_extension_point('RESPONSE_SHUTDOWN', $content, [], true));
210         }
211     }

213     /**
214      * Sends content to the client.
215      *
216      * @param string $content      Content
217      * @param string $contentType  Content type
218      * @param int    $lastModified HTTP Last-Modified Timestamp
219      * @param string $etag         HTTP Cachekey to identify the cache
220      */
221     public static function sendContent($content, $contentType = null, $lastModified = null, $etag = null)
222     {
223         if (!self::$sentContentType) {
224             self::sendContentType($contentType);
225         }
226         if (!self::$sentCacheControl) {
227             self::sendCacheControl();
228         }
229 
230         $environment = rex::isBackend() ? 'backend' : 'frontend';
231 
232         if (
233             self::$httpStatus == self::HTTP_OK &&
234             // Safari incorrectly caches 304s as empty pages, so don't serve it 304s
235             // http://tech.vg.no/2013/10/02/ios7-bug-shows-white-page-when-getting-304-not-modified-from-server/
236             // https://bugs.webkit.org/show_bug.cgi?id=32829
237             (!empty($_SERVER['HTTP_USER_AGENT']) && (false === strpos($_SERVER['HTTP_USER_AGENT'], 'Safari') || false !== strpos($_SERVER['HTTP_USER_AGENT'], 'Chrome')))
238         ) {
239             // ----- Last-Modified
240             if (!self::$sentLastModified
241                 && (rex::getProperty('use_last_modified') === true || rex::getProperty('use_last_modified') === $environment)
242             ) {
243                 self::sendLastModified($lastModified);
244             }
245 
246             // ----- ETAG
247             if (!self::$sentEtag
248                 && (rex::getProperty('use_etag') === true || rex::getProperty('use_etag') === $environment)
249             ) {
250                 self::sendEtag($etag ?: self::md5($content));
251             }
252         }
253 
254         // ----- GZIP
255         if (rex::getProperty('use_gzip') === true || rex::getProperty('use_gzip') === $environment) {
256             $content = self::sendGzip($content);
257         }
258 
259         self::cleanOutputBuffers();
260 
261         header('HTTP/1.1 ' . self::$httpStatus);
262 
263         // content length schicken, damit der browser einen ladebalken anzeigen kann
264         header('Content-Length: ' . rex_string::size($content));
265 
266         self::sendAdditionalHeaders();
267         self::sendPreloadHeaders();
268 
269         echo $content;
270 
271         if (function_exists('fastcgi_finish_request')) {
272             fastcgi_finish_request();
273         }
274     }

306     /**
307      * Checks if content has changed by the last modified timestamp.
308      *
309      * HTTP_IF_MODIFIED_SINCE feature
310      *
311      * @param int $lastModified HTTP Last-Modified Timestamp
312      */
313     public static function sendLastModified($lastModified = null)
314     {
315         if (!$lastModified) {
316             $lastModified = time();
317         }
318 
319         $lastModified = gmdate('D, d M Y H:i:s T', (float) $lastModified);
320 
321         // Sende Last-Modification time
322         header('Last-Modified: ' . $lastModified);
323 
324         // Last-Modified Timestamp gefunden
325         // => den Browser anweisen, den Cache zu verwenden
326         if (isset($_SERVER['HTTP_IF_MODIFIED_SINCE']) && $_SERVER['HTTP_IF_MODIFIED_SINCE'] == $lastModified) {
327             self::cleanOutputBuffers();
328 
329             header('HTTP/1.1 ' . self::HTTP_NOT_MODIFIED);
330             exit;
331         }
332         self::$sentLastModified = true;
333     }

335     /**
336      * Checks if content has changed by the etag cachekey.
337      *
338      * HTTP_IF_NONE_MATCH feature
339      *
340      * @param string $cacheKey HTTP Cachekey to identify the cache
341      */
342     public static function sendEtag($cacheKey)
343     {
344         // Laut HTTP Spec muss der Etag in " sein
345         $cacheKey = '"' . $cacheKey . '"';
346 
347         // Sende CacheKey als ETag
348         header('ETag: ' . $cacheKey);
349 
350         // CacheKey gefunden
351         // => den Browser anweisen, den Cache zu verwenden
352         if (isset($_SERVER['HTTP_IF_NONE_MATCH']) && $_SERVER['HTTP_IF_NONE_MATCH'] == $cacheKey) {
353             self::cleanOutputBuffers();
354 
355             header('HTTP/1.1 ' . self::HTTP_NOT_MODIFIED);
356             exit;
357         }
358         self::$sentEtag = true;
359     }

361     /**
362      * Encodes the content with GZIP/X-GZIP if the browser supports one of them.
363      *
364      * HTTP_ACCEPT_ENCODING feature
365      *
366      * @param string $content Content
367      *
368      * @return string
369      */
370     protected static function sendGzip($content)
371     {
372         $enc = '';
373         $encodings = [];
374         $supportsGzip = false;
375 
376         // Check if it supports gzip
377         if (isset($_SERVER['HTTP_ACCEPT_ENCODING'])) {
378             $encodings = explode(',', strtolower(preg_replace('/\s+/', '', $_SERVER['HTTP_ACCEPT_ENCODING'])));
379         }
380 
381         if ((in_array('gzip', $encodings) || in_array('x-gzip', $encodings) || isset($_SERVER['---------------']))
382             && function_exists('ob_gzhandler')
383             && !ini_get('zlib.output_compression')
384         ) {
385             $enc = in_array('x-gzip', $encodings) ? 'x-gzip' : 'gzip';
386             $supportsGzip = true;
387         }
388 
389         if ($supportsGzip) {
390             header('Content-Encoding: ' . $enc);
391             $content = gzencode($content, 9, FORCE_GZIP);
392         }
393 
394         return $content;
395     }

399     /**
400      * @param string      $name    The name of the cookie
401      * @param string|null $value   The value of the cookie, a empty value to delete the cookie.
402      * @param array       $options Different cookie Options. Supported keys are:
403      *                             "expires" int|string|\DateTimeInterface The time the cookie expires
404      *                             "path" string                           The path on the server in which the cookie will be available on
405      *                             "domain" string|null                    The domain that the cookie is available to
406      *                             "secure" bool                           Whether the cookie should only be transmitted over a secure HTTPS connection from the client
407      *                             "httponly" bool                         Whether the cookie will be made accessible only through the HTTP protocol
408      *                             "samesite" string|null                  Whether the cookie will be available for cross-site requests
409      *                             "raw" bool                              Whether the cookie value should be sent with no url encoding
410      *
411      * @throws \InvalidArgumentException
412      */
413     public static function sendCookie($name, $value, array $options = [])
414     {
415         $expire = isset($options['expires']) ? $options['expires'] : 0;
416         $path = isset($options['path']) ? $options['path'] : '/';
417         $domain = isset($options['domain']) ? $options['domain'] : null;
418         $secure = isset($options['secure']) ? $options['secure'] : false;
419         $httpOnly = isset($options['httponly']) ? $options['httponly'] : true;
420         $sameSite = isset($options['samesite']) ? $options['samesite'] : null;
421         $raw = isset($options['raw']) ? $options['raw'] : false;
422 
423         // from PHP source code
424         if (preg_match("/[=,; \t\r\n\013\014]/", $name)) {
425             throw new \InvalidArgumentException(sprintf('The cookie name "%s" contains invalid characters.', $name));
426         }
427         if (empty($name)) {
428             throw new \InvalidArgumentException('The cookie name cannot be empty.');
429         }
430         // convert expiration time to a Unix timestamp
431         if ($expire instanceof \DateTimeInterface) {
432             $expire = $expire->format('U');
433         } elseif (!is_numeric($expire)) {
434             $expire = strtotime($expire);
435             if (false === $expire) {
436                 throw new \InvalidArgumentException('The cookie expiration time is not valid.');
437             }
438         }
439 
440         $expire = 0 < $expire ? (int) $expire : 0;
441         $maxAge = $expire - time();
442         $maxAge = 0 >= $maxAge ? 0 : $maxAge;
443         $path = empty($path) ? '/' : $path;
444 
445         if (null !== $sameSite) {
446             $sameSite = strtolower($sameSite);
447         }
448         if (!in_array($sameSite, ['lax', 'strict', null], true)) {
449             throw new \InvalidArgumentException('The "sameSite" parameter value is not valid.');
450         }
451 
452         $str = 'Set-Cookie: '. ($raw ? $name : urlencode($name)).'=';
453         if ('' === (string) $value) {
454             $str .= 'deleted; expires='.gmdate('D, d-M-Y H:i:s T', time() - 31536001).'; Max-Age=0';
455         } else {
456             $str .= $raw ? $value : rawurlencode($value);
457             if (0 !== $expire) {
458                 $str .= '; expires='.gmdate('D, d-M-Y H:i:s T', $expire).'; Max-Age='.$maxAge;
459             }
460         }
461         if ($path) {
462             $str .= '; path='.$path;
463         }
464         if ($domain) {
465             $str .= '; domain='.$domain;
466         }
467         if ($secure) {
468             $str .= '; secure';
469         }
470         if ($httpOnly) {
471             $str .= '; httponly';
472         }
473         if ($sameSite) {
474             $str .= '; samesite='.$sameSite;
475         }
476 
477         header($str, false);
478     }

480     /**
481      * Creates the md5 checksum for the content.
482      *
483      * Dynamic content surrounded by `<!--DYN-->…<!--/DYN-->` is ignored.
484      *
485      * @param string $content
486      *
487      * @return string
488      */
489     private static function md5($content)
490     {
491         return md5(preg_replace('@<!--DYN-->.*<!--/DYN-->@U', '', $content));
492     }
