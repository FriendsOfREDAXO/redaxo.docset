 18     private static $httpStatus = self::HTTP_OK;
 19     private static $sentLastModified = false;

 26     /**
 27      * Sets the HTTP Status code.
 28      *
 29      * @param int $httpStatus
 30      *
 31      * @throws InvalidArgumentException
 32      */
 33     public static function setStatus($httpStatus)
 34     {
 35         if (strpos($httpStatus, "\n") !== false) {
 36             throw new InvalidArgumentException('Illegal http-status "' . $httpStatus . '", contains newlines');
 37         }
 38 
 39         self::$httpStatus = $httpStatus;
 40     }

 43      * Returns the HTTP Status code.
 44      *
 45      * @return string
 46      */
 47     public static function getStatus()
 48     {
 49         return self::$httpStatus;
 50     }

 52     /**
 53      * Set a http response header. A existing header with the same name will be overridden.
 54      *
 55      * @param string $name
 56      * @param string $value
 57      */
 58     public static function setHeader($name, $value)
 59     {
 60         self::$additionalHeaders[$name] = $value;
 61     }
 62 
 63     private static function sendAdditionalHeaders()
 64     {
 65         foreach (self::$additionalHeaders as $name => $value) {
 66             header($name .': ' . $value);
 67         }
 68     }

 70     /**
 71      * Set a file to be preload via http link header.
 72      *
 73      * @param $file
 74      * @param $type
 75      * @param $mimeType
 76      */
 77     public static function preload($file, $type, $mimeType)
 78     {
 79         self::$preloadFiles[] = [
 80             'file' => $file,
 81             'type' => $type,
 82             'mimeType' => $mimeType,
 83         ];
 84     }
 85 
 86     private static function sendPreloadHeaders()
 87     {
 88         foreach (self::$preloadFiles as $preloadFile) {
 89             header('Link: <' . $preloadFile['file'] . '>; rel=preload; as=' . $preloadFile['type'] . '; type="' . $preloadFile['mimeType'].'"; nopush', false);
 90         }
 91     }

 93     /**
 94      * Redirects to a URL.
 95      *
 96      * NOTE: Execution will stop within this method!
 97      *
 98      * @param string $url URL
 99      *
100      * @throws InvalidArgumentException
101      */
102     public static function sendRedirect($url)
103     {
104         if (strpos($url, "\n") !== false) {
105             throw new InvalidArgumentException('Illegal redirect url "' . $url . '", contains newlines');
106         }
107 
108         self::cleanOutputBuffers();
109         self::sendAdditionalHeaders();
110         self::sendPreloadHeaders();
111 
112         header('HTTP/1.1 ' . self::$httpStatus);
113         header('Location: ' . $url);
114         exit;
115     }

117     /**
118      * Sends a file to client.
119      *
120      * @param string $file               File path
121      * @param string $contentType        Content type
122      * @param string $contentDisposition Content disposition
123      */
124     public static function sendFile($file, $contentType, $contentDisposition = 'inline')
125     {
126         self::cleanOutputBuffers();
127 
128         if (!file_exists($file)) {
129             header('HTTP/1.1 ' . self::HTTP_NOT_FOUND);
130             exit;
131         }
132 
133         // prevent session locking while sending huge files
134         session_write_close();
135 
136         self::sendContentType($contentType);
137         header('Content-Disposition: ' . $contentDisposition . '; filename="' . basename($file) . '"');
138 
139         self::sendLastModified(filemtime($file));
140 
141         header('HTTP/1.1 ' . self::$httpStatus);
142         if (!self::$sentCacheControl) {
143             self::sendCacheControl('max-age=3600, must-revalidate, proxy-revalidate, private');
144         }
145 
146         // content length schicken, damit der browser einen ladebalken anzeigen kann
147         if (!ini_get('zlib.output_compression')) {
148             header('Content-Length: ' . filesize($file));
149         }
150 
151         self::sendAdditionalHeaders();
152         self::sendPreloadHeaders();
153 
154         readfile($file);
155     }

157     /**
158      * Sends a resource to the client.
159      *
160      * @param string $content      Content
161      * @param string $contentType  Content type
162      * @param int    $lastModified HTTP Last-Modified Timestamp
163      * @param string $etag         HTTP Cachekey to identify the cache
164      */
165     public static function sendResource($content, $contentType = null, $lastModified = null, $etag = null)
166     {
167         self::sendCacheControl('max-age=3600, must-revalidate, proxy-revalidate, private');
168         self::sendContent($content, $contentType, $lastModified, $etag);
169     }

171     /**
172      * Sends a page to client.
173      *
174      * The page content can be modified by the Extension Point OUTPUT_FILTER
175      *
176      * @param string $content      Content of page
177      * @param int    $lastModified HTTP Last-Modified Timestamp
178      */
179     public static function sendPage($content, $lastModified = null)
180     {
181         // ----- EXTENSION POINT
182         $content = rex_extension::registerPoint(new rex_extension_point('OUTPUT_FILTER', $content));
183 
184         $hasShutdownExtension = rex_extension::isRegistered('RESPONSE_SHUTDOWN');
185         if ($hasShutdownExtension) {
186             header('Connection: close');
187         }
188 
189         self::sendContent($content, null, $lastModified);
190 
191         // ----- EXTENSION POINT - (read only)
192         if ($hasShutdownExtension) {
193             // unlock session
194             session_write_close();
195 
196             rex_extension::registerPoint(new rex_extension_point('RESPONSE_SHUTDOWN', $content, [], true));
197         }
198     }

200     /**
201      * Sends content to the client.
202      *
203      * @param string $content      Content
204      * @param string $contentType  Content type
205      * @param int    $lastModified HTTP Last-Modified Timestamp
206      * @param string $etag         HTTP Cachekey to identify the cache
207      */
208     public static function sendContent($content, $contentType = null, $lastModified = null, $etag = null)
209     {
210         if (!self::$sentContentType) {
211             self::sendContentType($contentType);
212         }
213         if (!self::$sentCacheControl) {
214             self::sendCacheControl();
215         }
216 
217         $environment = rex::isBackend() ? 'backend' : 'frontend';
218 
219         if (
220             self::$httpStatus == self::HTTP_OK &&
221             // Safari incorrectly caches 304s as empty pages, so don't serve it 304s
222             // http://tech.vg.no/2013/10/02/ios7-bug-shows-white-page-when-getting-304-not-modified-from-server/
223             // https://bugs.webkit.org/show_bug.cgi?id=32829
224             (false === strpos($_SERVER['HTTP_USER_AGENT'], 'Safari') || false !== strpos($_SERVER['HTTP_USER_AGENT'], 'Chrome'))
225         ) {
226             // ----- Last-Modified
227             if (!self::$sentLastModified
228                 && (rex::getProperty('use_last_modified') === true || rex::getProperty('use_last_modified') === $environment)
229             ) {
230                 self::sendLastModified($lastModified);
231             }
232 
233             // ----- ETAG
234             if (!self::$sentEtag
235                 && (rex::getProperty('use_etag') === true || rex::getProperty('use_etag') === $environment)
236             ) {
237                 self::sendEtag($etag ?: self::md5($content));
238             }
239         }
240 
241         // ----- GZIP
242         if (rex::getProperty('use_gzip') === true || rex::getProperty('use_gzip') === $environment) {
243             $content = self::sendGzip($content);
244         }
245 
246         self::cleanOutputBuffers();
247 
248         header('HTTP/1.1 ' . self::$httpStatus);
249 
250         // content length schicken, damit der browser einen ladebalken anzeigen kann
251         header('Content-Length: ' . rex_string::size($content));
252 
253         self::sendAdditionalHeaders();
254         self::sendPreloadHeaders();
255 
256         echo $content;
257 
258         if (function_exists('fastcgi_finish_request')) {
259             fastcgi_finish_request();
260         }
261     }

293     /**
294      * Checks if content has changed by the last modified timestamp.
295      *
296      * HTTP_IF_MODIFIED_SINCE feature
297      *
298      * @param int $lastModified HTTP Last-Modified Timestamp
299      */
300     public static function sendLastModified($lastModified = null)
301     {
302         if (!$lastModified) {
303             $lastModified = time();
304         }
305 
306         $lastModified = gmdate('D, d M Y H:i:s T', (float) $lastModified);
307 
308         // Sende Last-Modification time
309         header('Last-Modified: ' . $lastModified);
310 
311         // Last-Modified Timestamp gefunden
312         // => den Browser anweisen, den Cache zu verwenden
313         if (isset($_SERVER['HTTP_IF_MODIFIED_SINCE']) && $_SERVER['HTTP_IF_MODIFIED_SINCE'] == $lastModified) {
314             self::cleanOutputBuffers();
315 
316             header('HTTP/1.1 ' . self::HTTP_NOT_MODIFIED);
317             exit;
318         }
319         self::$sentLastModified = true;
320     }

322     /**
323      * Checks if content has changed by the etag cachekey.
324      *
325      * HTTP_IF_NONE_MATCH feature
326      *
327      * @param string $cacheKey HTTP Cachekey to identify the cache
328      */
329     public static function sendEtag($cacheKey)
330     {
331         // Laut HTTP Spec muss der Etag in " sein
332         $cacheKey = '"' . $cacheKey . '"';
333 
334         // Sende CacheKey als ETag
335         header('ETag: ' . $cacheKey);
336 
337         // CacheKey gefunden
338         // => den Browser anweisen, den Cache zu verwenden
339         if (isset($_SERVER['HTTP_IF_NONE_MATCH']) && $_SERVER['HTTP_IF_NONE_MATCH'] == $cacheKey) {
340             self::cleanOutputBuffers();
341 
342             header('HTTP/1.1 ' . self::HTTP_NOT_MODIFIED);
343             exit;
344         }
345         self::$sentEtag = true;
346     }

348     /**
349      * Encodes the content with GZIP/X-GZIP if the browser supports one of them.
350      *
351      * HTTP_ACCEPT_ENCODING feature
352      *
353      * @param string $content Content
354      *
355      * @return string
356      */
357     protected static function sendGzip($content)
358     {
359         $enc = '';
360         $encodings = [];
361         $supportsGzip = false;
362 
363         // Check if it supports gzip
364         if (isset($_SERVER['HTTP_ACCEPT_ENCODING'])) {
365             $encodings = explode(',', strtolower(preg_replace('/\s+/', '', $_SERVER['HTTP_ACCEPT_ENCODING'])));
366         }
367 
368         if ((in_array('gzip', $encodings) || in_array('x-gzip', $encodings) || isset($_SERVER['---------------']))
369             && function_exists('ob_gzhandler')
370             && !ini_get('zlib.output_compression')
371         ) {
372             $enc = in_array('x-gzip', $encodings) ? 'x-gzip' : 'gzip';
373             $supportsGzip = true;
374         }
375 
376         if ($supportsGzip) {
377             header('Content-Encoding: ' . $enc);
378             $content = gzencode($content, 9, FORCE_GZIP);
379         }
380 
381         return $content;
382     }

384     /**
385      * Creates the md5 checksum for the content.
386      *
387      * Dynamic content surrounded by `<!--DYN-->…<!--/DYN-->` is ignored.
388      *
389      * @param string $content
390      *
391      * @return string
392      */
393     private static function md5($content)
394     {
395         return md5(preg_replace('@<!--DYN-->.*<!--/DYN-->@U', '', $content));
396     }
