  9 class rex_backend_login extends rex_login
 10 {
 11     const SYSTEM_ID = 'backend_login';
 12     const LOGIN_TRIES_1 = 3;
 13     const RELOGIN_DELAY_1 = 5;    // relogin delay after LOGIN_TRIES_1 tries
 14     const LOGIN_TRIES_2 = 50;
 15     const RELOGIN_DELAY_2 = 3600; // relogin delay after LOGIN_TRIES_2 tries
 16 
 17     private $tableName;
 18     private $stayLoggedIn;
 19 
 20     public function __construct()
 21     {
 22         parent::__construct();
 23 
 24         $tableName = rex::getTablePrefix() . 'user';
 25         $this->setSqlDb(1);
 26         $this->setSystemId(self::SYSTEM_ID);
 27         $this->setSessionDuration(rex::getProperty('session_duration'));
 28         $qry = 'SELECT * FROM ' . $tableName . ' WHERE status=1';
 29         $this->setUserQuery($qry . ' AND id = :id');
 30         // XXX because with concat the time into the sql query, users of this class should use checkLogin() immediately after creating the object.
 31         $this->setLoginQuery($qry . '
 32             AND login = :login
 33             AND (login_tries < ' . self::LOGIN_TRIES_1 . '
 34                 OR login_tries < ' . self::LOGIN_TRIES_2 . ' AND UNIX_TIMESTAMP(lasttrydate) < ' . (time() - self::RELOGIN_DELAY_1) . '
 35                 OR UNIX_TIMESTAMP(lasttrydate) < ' . (time() - self::RELOGIN_DELAY_2) . '
 36             )'
 37         );
 38         $this->tableName = $tableName;
 39     }
 40 
 41     public function setStayLoggedIn($stayLoggedIn = false)
 42     {
 43         $this->stayLoggedIn = $stayLoggedIn;
 44     }

 46     public function checkLogin()
 47     {
 48         $sql = rex_sql::factory();
 49         $userId = $this->getSessionVar('UID');
 50         $cookiename = self::getStayLoggedInCookieName();
 51 
 52         if ($cookiekey = rex_cookie($cookiename, 'string')) {
 53             if (!$userId) {
 54                 $sql->setQuery('SELECT id FROM ' . rex::getTable('user') . ' WHERE cookiekey = ? LIMIT 1', [$cookiekey]);
 55                 if ($sql->getRows() == 1) {
 56                     $this->setSessionVar('UID', $sql->getValue('id'));
 57                     rex_response::sendCookie($cookiename, $cookiekey, ['expires' => strtotime('+1 year'), 'samesite' => 'strict']);
 58                 } else {
 59                     self::deleteStayLoggedInCookie();
 60                 }
 61             }
 62             $this->setSessionVar('STAMP', time());
 63         }
 64 
 65         $check = parent::checkLogin();
 66 
 67         if ($check) {
 68             // gelungenen versuch speichern | login_tries = 0
 69             if ($this->userLogin != '' || !$userId) {
 70                 self::regenerateSessionId();
 71                 $params = [];
 72                 $add = '';
 73                 if ($this->stayLoggedIn || $cookiekey) {
 74                     $cookiekey = sha1($this->systemId . time() . $this->userLogin);
 75                     $add = 'cookiekey = ?, ';
 76                     $params[] = $cookiekey;
 77                     rex_response::sendCookie($cookiename, $cookiekey, ['expires' => strtotime('+1 year'), 'samesite' => 'strict']);
 78                 }
 79                 if (self::passwordNeedsRehash($this->user->getValue('password'))) {
 80                     $add .= 'password = ?, ';
 81                     $params[] = self::passwordHash($this->userPassword, true);
 82                 }
 83                 array_push($params, rex_sql::datetime(), rex_sql::datetime(), session_id(), $this->userLogin);
 84                 $sql->setQuery('UPDATE ' . $this->tableName . ' SET ' . $add . 'login_tries=0, lasttrydate=?, lastlogin=?, session_id=? WHERE login=? LIMIT 1', $params);
 85             }
 86 
 87             $this->user = new rex_user($this->user);
 88 
 89             if ($this->impersonator instanceof rex_sql) {
 90                 $this->impersonator = new rex_user($this->impersonator);
 91             }
 92         } else {
 93             // fehlversuch speichern | login_tries++
 94             if ($this->userLogin != '') {
 95                 $sql->setQuery('SELECT login_tries FROM ' . $this->tableName . ' WHERE login=? LIMIT 1', [$this->userLogin]);
 96                 if ($sql->getRows() > 0) {
 97                     $login_tries = $sql->getValue('login_tries');
 98                     $sql->setQuery('UPDATE ' . $this->tableName . ' SET login_tries=login_tries+1,session_id="",cookiekey="",lasttrydate=? WHERE login=? LIMIT 1', [rex_sql::datetime(), $this->userLogin]);
 99                     if ($login_tries >= self::LOGIN_TRIES_1 - 1) {
100                         $time = $login_tries < self::LOGIN_TRIES_2 ? self::RELOGIN_DELAY_1 : self::RELOGIN_DELAY_2;
101                         $hours = floor($time / 3600);
102                         $mins = floor(($time - ($hours * 3600)) / 60);
103                         $secs = $time % 60;
104                         $formatted = ($hours ? $hours . 'h ' : '') . ($hours || $mins ? $mins . 'min ' : '') . $secs . 's';
105                         $this->message .= ' ' . rex_i18n::rawMsg('login_wait', '<strong data-time="' . $time . '">' . $formatted . '</strong>');
106                     }
107                 }
108             }
109         }
110 
111         if ($this->isLoggedOut() && $userId != '') {
112             $sql->setQuery('UPDATE ' . $this->tableName . ' SET session_id="", cookiekey="" WHERE id=? LIMIT 1', [$userId]);
113             self::deleteStayLoggedInCookie();
114         }
115 
116         return $check;
117     }

119     public static function deleteSession()
120     {
121         self::startSession();
122 
123         unset($_SESSION[rex::getProperty('instname')][self::SYSTEM_ID]);
124         self::deleteStayLoggedInCookie();
125 
126         rex_csrf_token::removeAll();
127     }
128 
129     private static function deleteStayLoggedInCookie()
130     {
131         rex_response::sendCookie(self::getStayLoggedInCookieName(), '');
132     }
133 
134     private static function getStayLoggedInCookieName()
135     {
136         return 'rex_user_' . sha1(rex::getProperty('instname'));
137     }
138 
139     public static function hasSession()
140     {
141         // try to fast-fail, so we dont need to start a session in all cases (which would require a session lock...)
142         if (!isset($_COOKIE[session_name()])) {
143             return false;
144         }
145         self::startSession();
146 
147         $instname = rex::getProperty('instname');
148 
149         return isset($_SESSION[$instname][self::SYSTEM_ID]['UID']) && $_SESSION[$instname][self::SYSTEM_ID]['UID'] > 0;
150     }

152     /**
153      * Creates the user object if it does not already exist.
154      *
155      * Helpful if you want to check permissions of the backend user in frontend.
156      * If you only want to know if there is any backend session, use {@link rex_backend_login::hasSession()}.
157      *
158      * @return rex_user
159      */
160     public static function createUser()
161     {
162         if (!self::hasSession()) {
163             return null;
164         }
165         if ($user = rex::getUser()) {
166             return $user;
167         }
168 
169         $login = new self();
170         rex::setProperty('login', $login);
171         if ($login->checkLogin()) {
172             $user = $login->getUser();
173             rex::setProperty('user', $user);
174             return $user;
175         }
176         return null;
177     }
