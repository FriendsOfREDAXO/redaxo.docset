 22     {
 23         if (isset(self::$cssFiles[$media]) && in_array($file, self::$cssFiles[$media])) {

242     /**
243      * Returns the formatted title.
244      *
245      * @param string            $head
246      * @param null|string|array $subtitle
247      *
248      * @throws InvalidArgumentException
249      *
250      * @return string
251      */
252     public static function title($head, $subtitle = null)
253     {
254         if ($subtitle !== null && !is_string($subtitle) && (!is_array($subtitle) || count($subtitle) > 0 && !reset($subtitle) instanceof rex_be_page)) {
255             throw new InvalidArgumentException('Expecting $subtitle to be a string or an array of rex_be_page!');
256         }
257 
258         if ($subtitle === null) {
259             $subtitle = rex_be_controller::getPageObject(rex_be_controller::getCurrentPagePart(1))->getSubpages();
260         }
261 
262         if (is_array($subtitle) && count($subtitle) && reset($subtitle) instanceof rex_be_page) {
263             $nav = rex_be_navigation::factory();
264             $nav->setHeadline('default', rex_i18n::msg('subnavigation', $head));
265             foreach ($subtitle as $pageObj) {
266                 $nav->addPage($pageObj);
267             }
268             $blocks = $nav->getNavigation();
269             $navigation = [];
270             if (count($blocks) == 1) {
271                 $navigation = current($blocks);
272                 $navigation = $navigation['navigation'];
273             }
274 
275             if (!empty($navigation)) {
276                 $fragment = new rex_fragment();
277                 $fragment->setVar('left', $navigation, false);
278                 $subtitle = $fragment->parse('core/navigations/content.php');
279             } else {
280                 $subtitle = '';
281             }
282         } elseif (!is_string($subtitle)) {
283             $subtitle = '';
284         }
285 
286         $title = rex_extension::registerPoint(new rex_extension_point('PAGE_TITLE', $head));
287 
288         $fragment = new rex_fragment();
289         $fragment->setVar('heading', $title, false);
290         $fragment->setVar('subtitle', $subtitle, false);
291         $return = $fragment->parse('core/page/header.php');
292 
293         $return .= rex_extension::registerPoint(new rex_extension_point('PAGE_TITLE_SHOWN', ''));
294 
295         return $return;
296     }

298     /**
299      * Returns a clang switch.
300      *
301      * @param rex_context $context
302      * @param bool        $drop
303      *
304      * @return string
305      */
306     public static function clangSwitch(rex_context $context, $drop = true)
307     {
308         if (rex_clang::count() == 1) {
309             return '';
310         }
311 
312         if ($drop && rex_clang::count() >= 4) {
313             return self::clangSwitchAsDropdown($context);
314         }
315 
316         $items = [];
317         foreach (rex_clang::getAll() as $id => $clang) {
318             if (rex::getUser()->getComplexPerm('clang')->hasPerm($id)) {
319                 $icon = ($id == $context->getParam('clang')) ? '<i class="rex-icon rex-icon-language-active"></i> ' : '<i class="rex-icon rex-icon-language"></i> ';
320                 $item = [];
321                 $item['href'] = $context->getUrl(['clang' => $id]);
322                 $item['title'] = $icon . rex_i18n::translate($clang->getName());
323                 if ($id == $context->getParam('clang')) {
324                     $item['active'] = true;
325                 }
326                 $items[] = $item;
327             }
328         }
329         $fragment = new rex_fragment();
330         $fragment->setVar('left', $items, false);
331 
332         return $fragment->parse('core/navigations/content.php');
333     }

376      *

378      *

381     public static function clangSwitchAsDropdown(rex_context $context)
382     {

385         }
386 
387         $button_label = '';
388         $items = [];
389         foreach (rex_clang::getAll() as $id => $clang) {
390             if (rex::getUser()->getComplexPerm('clang')->hasPerm($id)) {
391                 $item = [];
392                 $item['title'] = rex_i18n::translate($clang->getName());
393                 $item['href'] = $context->getUrl(['clang' => $id]);
394                 if ($id == $context->getParam('clang')) {
395                     $item['active'] = true;
396                     $button_label = rex_i18n::translate($clang->getName());
397                 }
398                 $items[] = $item;
399             }
400         }
401 
402         $fragment = new rex_fragment();
403         $fragment->setVar('class', 'rex-language');
404         $fragment->setVar('button_prefix', rex_i18n::msg('language'));
405         $fragment->setVar('button_label', $button_label);
406         $fragment->setVar('header', rex_i18n::msg('clang_select'));
407         $fragment->setVar('items', $items, false);
408 
409         if (rex::getUser()->isAdmin()) {
410             $fragment->setVar('footer', '<a href="' . rex_url::backendPage('system/lang') . '"><i class="fa fa-flag"></i> ' . rex_i18n::msg('languages_edit') . '</a>', false);
411         }
412 
413         return $fragment->parse('core/dropdowns/dropdown.php');
